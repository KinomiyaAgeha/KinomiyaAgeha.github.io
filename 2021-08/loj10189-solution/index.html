#include<cstdio>
#include<iostream>
using namespace std;
#define R register
#define ll long long
const int N=1e6+10;
int n, l, r, x[N], p[N], c[N], q[N];
ll sp[N], s[N], f[N];
double calc(int x,int y) { return 1.0*(f[x]-f[y]+s[x]-s[y])/(sp[x]-sp[y]); }
/* 不用double在某些OJ精度会炸*/
int main() {
    R int i;
    scanf("%d",&n);
    for(i=1;i<=n;++i) {
        scanf("%d%d%d",&x[i],&p[i],&c[i]);
        sp[i]=sp[i-1]+p[i], s[i]=s[i-1]+(ll)x[i]*p[i];
    }
    for(i=1;i<=n;++i) {
        while(l<r&&calc(q[l+1],q[l])<x[i]) ++l;
        f[i]=f[q[l]]+(ll)x[i]*(sp[i]-sp[q[l]])-s[i]+s[q[l]]+c[i];
        while(l<r&&calc(q[r],q[r-1])>calc(i,q[r])) --r;
        q[++r]=i;
    }
    ll ans=0x7fffffffffffffffll;
    while(!p[n]) ans=min(ans,f[n--]);
    printf("%lld\n",min(ans,f[n]));
    /* 最后三行是防luogu上的hack */
}
