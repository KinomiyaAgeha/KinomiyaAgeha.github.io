luogu2480 古代猪文 题解

发表于2021-08-25|题解

link

求 显然的，那一长串的东西是质数。

设它为好了。

则有

由欧拉定理的推论 其中 与 互质，且 为任意整数。

可得 所以本题就变成了求右边的那个东西。

显然不是个质数。

不难发现但很难算出 可以用的时间枚举的约数。

考虑到 ，我们用定理求出，并记录它们对上述四个因子取模的结果，这样就得到了一个线性同余方程组。 然后用中国剩余定理求出满足上述条件的最小的 ，用快速幂求 即可。

注意代码细节。

#include<cstdio>
#include<iostream>
using namespace std;
#define R register
#define ll long long
const ll a[4]={2,3,4679,35617};
ll n, g, ans, p=999911658ll, b[4], fac[36000]={1};
ll fp(ll x,ll y) {
    R ll z=1;
    for(;y;x=x*x%p,y>>=1ll) if(y&1ll) z=z*x%p;
    return z;
}
void exgcd(ll a,ll b,ll& x,ll& y) {
    if(!b) { x=1, y=0; return ; }
    exgcd(b,a%b,y,x), y-=a/b*x;
}
ll inv(ll a,ll p) {
    ll x, y;
    exgcd(a,p,x,y);
    return (x%p+p)%p;
}
ll C(ll x,ll p) {
    ll z=1, y=n;
    for(;x;x/=p,y/=p) {
        ll a=x%p, b=y%p;
        z=z*fac[b]%p*inv(fac[a],p)%p*inv(b<a?0:fac[b-a],p)%p;
    }
    return z;
}
int main() {
    scanf("%lld%lld",&n,&g), g%=p+1;
    if(!g) { puts("0"); return 0; }
    for(int i=1;i<=a[3];++i) fac[i]=fac[i-1]*i%p;
    for(int i=1;i*i<=n;++i) if(n%i==0) {
        for(int j=0;j<4;++j) b[j]=(b[j]+C(i,a[j]))%a[j];
        if(i*i!=n) for(j=0;j<4;++j) b[j]=(b[j]+C(n/i,a[j]))%a[j];
    }
    for(int i=0;i<4;++i) {
        ll x, y;
        exgcd(p/a[i],a[i],x,y);
        ans=(ans+x*(p/a[i])%p*b[i])%p;
    }
    ans=(ans+p)%p, ++p;
    printf("%lld\n",fp(g,ans));
}
