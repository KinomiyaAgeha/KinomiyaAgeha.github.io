这时候就要用上大毒瘤欧拉的欧拉函数了！

由于欧拉函数是与一个数互质，那么用前缀和。

由于 与 算两种，所以计数时要乘2，但是会多加一次 ，减去就好了。

设 为 的约数个数， 为 的第 个约数。

实现的时候用欧拉筛，更加毒瘤了

#include<cstdio>
#include<iostream>
using namespace std;
#define ll long long
const int N=1e7+6;
ll n, m, ans, v[N], p[N], phi[N];
int main() {
    int i, j;
    scanf("%lld",&n);
    phi[1]=1;
    for(i=2;i<=n;++i) {
        if(!v[i]) v[i]=i, p[++m]=i, phi[i]=i-1;
        for(j=1;j<=m;++j) {
            if(p[j]>v[i]||p[j]*i>n) break ;
            v[p[j]*i]=p[j];
            phi[p[j]*i]=phi[i]*(i%p[j]?p[j]-1:p[j]);
        }
        phi[i]+=phi[i-1];
    }
    for(i=1;i<=m;++i) ans+=phi[n/p[i]];
    printf("%lld\n",(ans<<1)-m);
}
