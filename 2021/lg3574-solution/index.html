luogu3574 FarmCraft 题解

发表于2021-10-16|题解

树形 DP。

设 为完成以 为根的配送的最短时间。

引入 作为遍历以 为根的子树并回到 ，所需要的时间。

首先考虑 的求法，因为每遍历完一棵子树都要回到 ，所以有 而 对求出 有用，当且仅当 处于 的前一棵子树的阶段。

具体地说，设 子节点集合为 ，则 转移为 是因为 表示的状态已经回到了 ，但还没有到 表示的 ，所以要加上一条路的时间。

而取最大值，则是配置电脑可以在配送的同时进行，如果能处理完时间更长的状态，那么时间更短的状态一定被处理完了。

特别的，当时，不能先在比特安萨尔处安装，所以 答案 如何实现有顺序地遍历子树呢？贪心。

用邻项交换证明，此处省略。

按照 递减排序。

理论复杂度 $ O(nk _2 k)kson(x)$ 的平均值。

实际比低很多，可以 AC 本题。

#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
#define R register
const int N=5e5+5;
int n, k, c[N], f[N], g[N], u[N];
int cnt, h[N], ver[N<<1], nxt[N<<1];
void add(int x,int y) { ver[++cnt]=y, nxt[cnt]=h[x], h[x]=cnt; }
bool cmp(int x,int y) { return g[x]-f[x]<g[y]-f[y]; }
void dp(int x,int fr) {
    R int i, y, k=0;
    if(x!=1) f[x]=c[x];
    for(i=h[x];i;i=nxt[i]) if(ver[i]!=fr) dp(ver[i],x);
    for(i=h[x];i;i=nxt[i]) if(ver[i]!=fr) u[++k]=ver[i];
    sort(u+1,u+k+1,cmp);
    for(i=1;i<=k;++i) {
        y=u[i];
        f[x]=max(f[x],f[y]+g[x]+1), g[x]+=g[y]+2;
    }
}
int main() {
    R int i, x, y;
    scanf("%d",&n);
    for(i=1;i<=n;++i) scanf("%d",&c[i]);
    for(i=1;i<n;++i) scanf("%d%d",&x,&y), add(x,y), add(y,x);
    dp(1,0);
    printf("%d\n",max(f[1],g[1]+c[1]));
}
