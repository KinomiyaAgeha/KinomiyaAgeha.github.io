LOJ2652 Zap 题解

发表于2021-08-25|题解

link

题目要求等价于求满足 并且 的二元组 的数量。

设 为满足 且 的二元组 的数量，不难想到 由容斥原理，得由于 ，比真正的答案多了 是 的倍数的个数，所以要减去它们，即 与 。

而减去后多减了 既是 的倍数，又是 的倍数的个数，即 ，又要加上。以此类推，我们发现加减的关系正是 的莫比乌斯函数。

记为 (x)$ 。

所以答案为 由于

所以实现的时候求出 的前缀和，直接处理一个区间的信息。

#include<cstdio>
#include<iostream>
using namespace std;
#define ll long long
const int N=5e4+10;
int n, a, b, k, Mobius[N];
bool v[N];
int sol() {
    int x, gx, ans=0;
    a/=k, b/=k;
    if(a>b) swap(a,b);
    for(x=1;x<=a;x=gx+1) {
        gx=min(a/(a/x),b/(b/x));
        ans+=(Mobius[gx]-Mobius[x-1])*(a/x)*(b/x);
    }
    return ans;
}
int main() {
    int i, j;
    for(i=1;i<=5e4;++i) Mobius[i]=1;
    for(i=2;i<=5e4;++i) if(!v[i]) {
        Mobius[i]=-1;
        for(j=(i<<1);j<=5e4;j+=i) {
            v[j]=1;
            if(j/i%i==0) Mobius[j]=0; else Mobius[j]*=-1;
        } 
    }
    for(i=1;i<=5e4;++i) Mobius[i]+=Mobius[i-1];
    scanf("%d",&n);
    while(n --> 0) {
        scanf("%d%d%d",&a,&b,&k);
        printf("%d\n",sol());
    }
}
