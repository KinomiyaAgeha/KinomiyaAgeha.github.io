T1

田忌赛马

秒打了个错误的 DP。

后来想了一下，排序后用双指针比较，贪心地选择即可。

#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
#define R register
const int N=2006;
int n, ans, a[N], b[N];
bool v[N];
bool cmp(int x,int y) { return x>y; }
int main() {
    R int i, j, x, y;
    scanf("%d",&n);
    for(i=1;i<=n;++i) scanf("%d",&a[i]);
    for(i=1;i<=n;++i) scanf("%d",&b[i]);
    sort(a+1,a+n+1,cmp), sort(b+1,b+n+1,cmp);
    for(i=1,j=1,x=n,y=n;i<=x;) {
        if(a[i]>b[j]) ans+=200, ++i, ++j;
        else if(a[i]<b[j]) ans-=200, --x, ++j;
        else {
            if(a[x]>b[y]) ans+=200, --x, --y;
            else {
                if(a[x]<b[j]) ans-=200;
                --x, ++j;
            }
        }
    }
    printf("%d\n",ans);
}

 

T2

爆弹虐场

一开始看错条件了。

「至少 个」看成了「个」。

于是想了一个错误的贪心加边的算法。

看清题目后，发现这题完全可以当最小生成树来作。有一个贪心将边排序的过程。

kruskal 算法结束后判断时候有 个并判断是否联通（是一棵树）即可。

#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
#define R register
const int N=10010;
int n, m, k, tk, ans, l, r=1e6+10, f[N];
struct E { int x, y, z, s; } a[N<<1];
bool cmp(E a,E b) { return a.z!=b.z? a.z<b.z:a.s<b.s; }
bool v[N];
int get(int x) { return x==f[x]? x:f[x]=get(f[x]); }
bool C(int mid) {
    R int i, x, y, z, s, tk=0, e=0;
    for(i=1;i<=n;++i) f[i]=i;
    sort(a+1,a+m+1,cmp);
    for(i=1;i<=m;++i) {
        x=get(a[i].x), y=get(a[i].y), z=a[i].z, s=a[i].s;
        if(min(z,s)>mid) continue ;
        if(x!=y) {
            if(z<=mid) ++tk;
            ++e, f[x]=y;
        }
    }
    return tk>=k&&e==n-1;
}
int main() {
    R int i;
    scanf("%d%d%d",&n,&k,&m);
    for(i=1;i<=m;++i)
        scanf("%d%d%d%d",&a[i].x,&a[i].y,&a[i].z,&a[i].s);
    while(l<=r) {
        int mid=l+r>>1;
        if(C(mid)) r=mid-1, ans=mid; else l=mid+1;
    }
    printf("%d\n",ans);
}

int lca(int x,int y) {
    R int i, j;
    if(d[x]<d[y]) swap(x,y);
    for(i=0,j=d[x]-d[y];j;j>>=1,++i) if(j&1) x=f[x][i];
    if(x==y) return x;
    for(i=lg[x];i>=0;--i) if(f[x][i]!=f[y][i]) x=f[x][i], y=f[y][i];
    /* lg[x] 是预处理的 log2(x)，这里应该写 lg[d[x]] */
    return f[x][0]; 
}
