CF1265E Beautiful Mirrors

solution 1

套路地设 为从 问到第 个镜子，且通过了第 个镜子的期望天数。

设

预处理 的逆元即可。

CODE

#include<bits/stdc++.h>
using namespace std;
#define int long long
const int N=2e5+5, P=100, mod=998244353;
int t, n, m, p[N], inv[N], f[N];
int read() {
	int a=0, f=1; char c=getchar();
	while(!isdigit(c)) {
		if(c=='-') f=-1;
		c=getchar();
	}
	while(isdigit(c)) a=a*10+c-'0', c=getchar();
	return a;
}
signed main() {
	n=read();
	for(int i=1;i<=n;++i) p[i]=read();
	inv[0]=inv[1]=1;
	for(int i=2;i<=100;++i) inv[i]=(mod-mod/i)*inv[mod%i]%mod;
	for(int i=1;i<=n;++i) {
		f[i]=(f[i-1]+1)%mod*P%mod*inv[p[i]]%mod;
	}
	printf("%lld\n",f[n]);
}

#include<bits/stdc++.h>
using namespace std;
#define int long long
const int N=1e6+5, mod=998244353;
int id, n, m, f[N], g[N], deg[N];
int tot, h[N], to[N<<1], nxt[N<<1];
bool v[N];
void add(int x,int y) { to[++tot]=y, nxt[tot]=h[x], h[x]=tot; }
int read() {
	int a=0, f=1; char c=getchar();
	while(!isdigit(c)) {
		if(c=='-') f=-1;
		c=getchar();
	}
	while(isdigit(c)) a=a*10+c-'0', c=getchar();
	return a;
}
signed main() {
	id=read(), n=read(), m=read();
	for(int i=1;i<=m;++i) {
		int x=read(), y=read();
		add(x,y), ++deg[x];
	}
	for(int x=1;x<=n;++x) {
		f[x]=deg[x]+1;
		for(int i=h[x];i;i=nxt[i]) {
			int y=to[i];
			(f[x]+=(g[x-1]-g[y-1]+mod)%mod)%=mod;
		}
		g[x]=(g[x-1]+f[x])%mod;
	}
	printf("%lld\n",g[n]);
}
