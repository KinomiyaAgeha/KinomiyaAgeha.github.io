CF1334E Divisor Paths 题解

注意到对于给定的两个点 ，若 并且 ，那么二者之间的最短路上节点一定是先减小再增加，换句话说，一定是分别从 和 出发，最终汇于一点。否则

考虑 的过程，无论怎样的走法，一定恰好消去了 。而每一条边只会消去一个质数，所以将 分解 每一个方案，等价于 的排列方式。而 有 个，根据可重集的全排列，得到方案数为 对于 ，同样处理即可。最终答案为

#include<bits/stdc++.h>
using namespace std;
#define int long long
int read() {
	int a=0, f=1; char c=getchar();
	while(!isdigit(c)) {
		if(c=='-') f=-1;
		c=getchar();
	}
	while(isdigit(c)) a=a*10+c-'0', c=getchar();
	return a*f;
}
const int mod=998244353;
int D, q, cnt, d[1000], fac[100];
// 虽然D很大，但是质因数个数并不会很多，质因数指数的阶乘就更少了
int fp(int a,int b) {
	int c=1;
	for(;b;a=a*a%mod,b>>=1) if(b&1) c=c*a%mod;
	return c;
}
int calc(int x) {
	int p=0, q=1;
	for(int i=1;i<=cnt;++i) if(x%d[i]==0) {
		int e=0;
		while(x%d[i]==0) x/=d[i], ++e;
		p+=e, (q*=fac[e])%=mod;
	}
	return fac[p]*fp(q,mod-2)%mod;
}
int gcd(int x,int y) { return y? gcd(y,x%y):x; }
signed main() {
	D=read(), q=read();
	for(int i=2;i*i<=D;++i) if(D%i==0) {
		while(D%i==0) D/=i;
		d[++cnt]=i;
	}
	if(D>1) d[++cnt]=D;
	fac[0]=fac[1]=1;
	for(int i=2;i<=100;++i) fac[i]=fac[i-1]*i%mod;
	while(q--) {
		int x=read(), y=read();
		int z=gcd(x,y);
		printf("%lld\n",calc(x/z)*calc(y/z)%mod);
	}
}
