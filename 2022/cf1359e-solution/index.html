CF1359E Modular Stability 题解

发表于2022-04-10|题解

由于 时最小的，所以对于长度为 的序列 ，取模每一项最后的结果是 。

而对于下标的任意排列，显然 ，都必须满足 ，即 ，不然最后的结果就会改变。当 为首项时，就得到了 。

所以，我们只要枚举首项 ，在 范围 的倍数个数为 ，我们要从中选取 个数作为其他项，所以答案就是 统计就好了。

#include<cstdio>
#define R register
#define ll long long
#define rep(i,j,k) for(i=(j);i<=(k);++i)
const int N=5e5+5;
const ll p=998244353;
int n, k;
ll fac[N], inv[N], ans;
ll C(ll n,ll m) {
    if(!m) return 1;
    return fac[n]*inv[m]%p*inv[n-m]%p;
}
int main() {
    R int i;
    scanf("%d%d",&n,&k);
    fac[0]=1ll, inv[0]=inv[1]=1ll;
    rep(i,1,5e5) {
        fac[i]=fac[i-1]*i%p;
        if(i!=1) inv[i]=(p-p/i)*inv[p%i]%p;
    }
    rep(i,1,5e5) inv[i]=inv[i]*inv[i-1]%p;
    rep(i,1,n) {
        if(n/i<k) break;
        (ans+=C(n/i-1,k-1))%=p;
    }
    printf("%lld\n",ans);
}
