#include<bits/stdc++.h>
using namespace std;
#define ll long long
long long T, n, a, b;
void solve() {
    scanf("%lld%lld%lld",&n,&a,&b);
    if(a==1) {
        if(n%b==1||b==1) puts("Yes");
        else puts("No");
        return;
    }
    int y=n%b;
    for(ll x=1;x<=n;x*=a) {
        if(x%b==y) { puts("Yes"); return; }
    }
    puts("No");
}
int main() {
    scanf("%lld",&T);
    while(T--) solve();
}
