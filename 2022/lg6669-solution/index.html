luogu6669 组合数问题 题解

在本题中 和 特别大，而质数 特别小，所以可以使用 Lucas 定理化简。注意，这里的 与 只是用两个数举例。 但是一次化简后 和 依然很大，所以我们要再用 Lucas 定理不断化简它。最后这个同余式就变成了 其中 ，同理。

为什么是这样的？因为我们让 与 不断取模并缩小的过程中就相当于把它们转化成了 进制，每一个常数项都是一位，上式实际是用这个 进制数表示 与 。

这样的话，要想使它为 0，只需要其中一项为 0 即可。更进一步的说，存在 。

所以如果一组 合法，那么把 像上面一样转化成 进制，就一定有上面的情况。

用数位 DP 求出所有不能被 整除的 ，最后用总数量减去就是答案。

按照转化成 进制后的位数去计算。设 为第 位，其中 中的 是否达到上限的方案数。

转移不太好写，见代码。

总方案数 可以理解为，当 中的 为 0 时有 1 种，为 1 时有 2 种，为 2 时有 3种，为 m 时有 m+1 种。求和就是第一项了。当 时，因为 所以都只有 种取值，就是第二项。

CODE

#include<bits/stdc++.h>
using namespace std;
#define ll long long
const int N=100;
const ll mod=1e9+7, inv=500000004; // inv是2%(10^9+7)的逆元
int t, k, c1, c2, a[N], b[N];
ll n, m, S, f[N][2][2];
ll dp(int len,int c,int d) {
    // len是k进制下的位数
	if(!len) return 1ll; // 边界，没有位数，只能是0
	if(f[len][c][d]!=-1) return f[len][c][d]%mod;
	ll F=0;
	int l=c? a[len]:k-1, r=d? b[len]:k-1;
    // 如果上一位没有达到边界，那么这一位可以随便取，由于是k进制下，所以边界是k-1
    // 如果达到边界了，这一位最多就取到这一位的数字上。
    // 边界，指的是读入的n和m在k进制下的第len位。
	for(int i=0;i<=l;++i) for(int j=0;j<=i&&j<=r;++j) {
        // i,j就是C(i,j)中的i,j，保证j<=i
		(F+=dp(len-1,c&&(i==l),d&&(j==r)))%=mod;    
        // 只有一路从dp(c1,1,1)递归过来c和d才取1
        // 所以如果c或d=1，那么一定是达到了n和m的k进制中第len-1位，也就是a和b数组
	}
	return f[len][c][d]=F%mod;
}
void sol() {
	c1=c2=0;
	memset(a,0,sizeof(a)), memset(b,0,sizeof(b));
	memset(f,-1,sizeof(f));
	scanf("%lld%lld",&n,&m);
	m=min(n,m); // 保证m<=n
	ll S1=(((m+1)%mod)*((m+2)%mod)%mod*inv)%mod;
	ll S2=(((n-m)%mod)*((m+1)%mod)%mod)%mod;
	S=(S1+S2)%mod;
    // 一定注意，每一次运算都要取模，不然……
	while(n) a[++c1]=n%k, n/=k;
	while(m) b[++c2]=m%k, m/=k;
    // 转化k进制
	printf("%lld\n",(S-dp(c1,1,1)+mod)%mod);
}
int main() {
	scanf("%d%d",&t,&k);
	while(t--) sol();
}
