luogu7716 Covering 题解

发表于2022-11-12|题解

设 为前 个纸片 ，用了 个的方案数。

对于一个纸片 ，如果它没有被覆盖过，那么直接继承 。

否则如果它没有被完全覆盖，设 为它所在的位置的四个方向上，编号大于它的个数。它可以把自己的另一半放到 个位置上。 如果被完全覆盖，那么可以不放它，也可以放到编号大于 的两个位置下面，设 为有多少能防止一个纸片的位置，满足两个格子编号大于 。 实现的时候注意几个细节。出现在棋盘上的编号是必须选取的，因此设 为当前棋盘已经出现了几个纸片，转移的时候强制从 开始。

可以预处理，再做一个后缀和。

CODE

const int N=1e3+5, mod=1e9+7;
int t, n, m, k, l, r, a[N][N], s[N], f[N][N];
pair<int,int> v[N];
void solve() {
	#define SET(x,y) memset(x,y,sizeof(x))
	SET(a,0), SET(s,0), SET(f,0), SET(v,0);
	n=read(), m=read(), k=read(), l=read(), r=read();
	int K=0, J=0;
	for(int i=1;i<=n;++i) for(int j=1;j<=m;++j) {
		a[i][j]=read();
		K=max(K,a[i][j]);
		v[a[i][j]]=make_pair(i,j);
		if(i>1&&a[i-1][j]) ++s[min(a[i][j],a[i-1][j])];
		if(j>1&&a[i][j-1]) ++s[min(a[i][j],a[i][j-1])];
	}
	for(int i=k;i;--i) (s[i]+=s[i+1])%=mod;
	f[0][0]=1;
	for(int i=1;i<=K;++i) {
		f[i][J]=f[i-1][J];
		if(v[i].first) {
			int x=v[i].first, y=v[i].second;
			if(a[x-1][y]==i||a[x+1][y]==i||a[x][y-1]==i||a[x][y+1]==i) {
				for(int j=J+1;j<=min(i,r);++j) f[i][j]=f[i-1][j-1];
				++J;
				continue;
			}
			int tag=(a[x-1][y]>i)+(a[x+1][y]>i)+(a[x][y-1]>i)+(a[x][y+1]>i);
			for(int j=J+1;j<=min(i,r);++j) f[i][j]=tag*f[i-1][j-1]%mod;
			++J;
		} else {
			for(int j=J+1;j<=min(i,r);++j) f[i][j]=(f[i-1][j]+s[i]*f[i-1][j-1]%mod)%mod;
		}
	}
	int ans=0;
	for(int i=max(J,l);i<=r;++i) (ans+=f[K][i])%=mod;
	printf("%lld\n",ans);
	
}
signed main() {
	t=read();
	while(t--) solve();
}
