luogu8818 策略游戏 题解

发表于2022-11-12|题解

分析

经典先手想要最大化，后手想要最小化。

我们不必拘泥于各种分类讨论，仔细思考不难发现，对于每一方，只有 4 种决策：最大正数，最小正数，最大负数，最小负数。枚举先手的情况，如果选择合法，那么就枚举后手的情况，将乘积取最小值，最后将所有先手的情况取个最大即可。

CODE

#include<bits/stdc++.h>
using namespace std;
#define int long long
#define SET(a,b) memset(a,b,sizeof(a))
int read() {
	int a=0, f=1; char c=getchar();
	while(!isdigit(c)) {
		if(c=='-') f=-1;
		c=getchar();
	}
	while(isdigit(c)) a=a*10+c-'0', c=getchar();
	return a*f;
}
const int N=1e5+5, inf=1e18;
int n, m, q, a[N], b[N], lg[N];
struct node { int v[4]; };
struct ST {
	node f[N][22];
	node gg(node x,node y) {
		node t;
		t.v[0]=max(x.v[0],y.v[0]);
		t.v[1]=max(x.v[1],y.v[1]);
		t.v[2]=min(x.v[2],y.v[2]);
		t.v[3]=min(x.v[3],y.v[3]);
		return t;
	}
	void init(int n,int* a) {
		for(int i=1;i<=n;++i) {
			if(a[i]>=0) f[i][0]={a[i],-inf,a[i],inf};
			else f[i][0]={-inf,a[i],inf,a[i]};
		}
		for(int j=1;(1<<j)<=n;++j)
			for(int i=1;i<=n-(1<<j)+1;++i)
				f[i][j]=gg(f[i][j-1],f[i+(1<<(j-1))][j-1]);
	}
	node query(int l,int r) {
		int k=lg[r-l+1];
		return gg(f[l][k],f[r-(1<<k)+1][k]);
	}
} L, Q;
bool check(int x) { return x>-inf&&x<inf; }
signed main() {
//	freopen("d:\\data\\game\\game3.in","r",stdin);
	n=read(), m=read(), q=read();
	for(int i=1;i<=n;++i) a[i]=read();
	for(int i=1;i<=m;++i) b[i]=read();
	for(int i=2;i<=max(n,m);++i) lg[i]=lg[i/2]+1;
	L.init(n,a), Q.init(m,b);
	while(q--) {
		int l1=read(), r1=read(), l2=read(), r2=read();
		node A=L.query(l1,r1);
		node B=Q.query(l2,r2);
		int ans=-inf;
		for(int i=0;i<4;++i) {
			if(check(A.v[i])) {
				int t=inf;
				for(int j=0;j<4;++j) if(check(B.v[j])) t=min(t,A.v[i]*B.v[j]);
				ans=max(ans,t);
			}
		}
		printf("%lld\n",ans);
	}
}
