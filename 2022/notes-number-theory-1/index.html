void exCRT() {
	M=m[1], R=r[1];
    // m[]是模数，r[]是同余数
	for(int i=2;i<=n;++i) {
		int d=r[i]-R, g, mod, x, y;
		g=exgcd(M,m[i],x,y);
        // 这里求解的是 Mx+m[i]y=gcd(M,m[i])
        // 根据等式的性质，易得等价于上文中的 p1x1+p2x2=1
		if(d%g) { ans=-1; return; }
        // 无解
		mod=m[i]/g;
        // 取模是因为要求最小正整数解，mod为什么这样算详见exgcd
		x=((x*d/g)%mod+mod)%mod;
        // 解
		R=x*M+R;
        // 更新R
		M=(M*m[i])/g;
        // M取lcm
	}
	ans=R;
    // 答案
}
