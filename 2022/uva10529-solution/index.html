#include<cstdio>
#include<iostream>
using namespace std;
const int N=1010;
int n;
double pl, pr, f[N];
double calc(int i,int j) { return (f[j-1]*(1-pr)+f[i-j]*(1-pl)+1)/(1-pl-pr); }
// 这个东西可不是单调的，但是有最小值，所以下面要选择calc(i,j)最小的j
void sol() {
	scanf("%lf%lf",&pl,&pr);
	for(int i=1,j=1;i<=n;++i) {
		while(j<i&&calc(i,j)>calc(i,j+1)) ++j;
		f[i]=calc(i,j);
	}
	printf("%.2lf\n",f[n]);
}
int main() { while(scanf("%d",&n)&&n) sol(); }
