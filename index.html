CF1179D Fedor Runs for President 题解

发表于 2021-10-30 更新于 2021-11-13 分类于 题解 阅读次数：

树上任意两点间，有且仅有一条简单路径。

数量
$$
\large C^2_n = \frac{n (n-1)}{2}
$$
对于添加边 $\large (x,y)$，设它们树上路径一点为 $\large z$。

加边后显然会多出一个简单环

设为 $\large S_z$ 为以 $\large z$ 为根且不含环上节点的子树大小。

则贡献为
$$
\large \frac{S_z(n-S_z)}{2}
$$
总贡献
$$
\large \frac{\sum_{z \in (x,y)} S_z(n-S_z)}{2}
\
\large = \frac{\sum_{z \in (x,y)} (S_zn-S_z^2)}{2}
\
\large = \frac{n \cdot \sum_{z \in (x,y)} S_z - \sum_{z \in (x,y) } S_z^2 }{2}
$$
考虑我们对 $\large S$ 的定义中，只含一个路径上的点。由于这条限制，所以
$$
\large \forall z_1,z_2 \in (x,y) \quad S_{z_1} \cap S_{z_2} = \phi
$$
所以得到总贡献
$$
\large \sum_{z \in (x,y)} S_z = n \Longrightarrow \frac{n^2 - \sum_{z \in (x,y) } S_z^2 }{2}
$$
问题转化为最小化 $\large \sum_{z \in (x,y) } S_z^2 = \Delta$，可以 DP。

 

先考虑 $\large S_z$ 如何求出。

设 $\large sz_x$ 以 $\large x$ 为根的子树大小，画图不难发现
$$
\large S_y = sz_x-sz_y \quad y \in son(x)
$$
懒得证了，挺显然的。

设 $\large f_x$ 为 添加路径是 $\large x$ 到以 $\large x$ 为根子树内任意一点，满足上式最小的贡献。
$$
\large f_x = \min_{y \in son(x)} { { f_y + (sz_x-sz_y) } }
$$
 

这样就可以计算 $\large \Delta$ 了。
$$
\large \Delta = \min_{y,z \in son(x)} { { f_y+f_z+(n-sz_y-sz_z)^2 }}
$$
显然可以斜率优化，但问题在于有两个需要维护的值。

不难想到，我们贪心地选择 $\large y$，维护最优的 $\large z$。

然后就是套路推式子。

但是个人感觉本题并不容易将转移变形成一次函数的形式，直接套两个决策 $\large z_1,z_2$ 就行，最后推出
$$
\large \frac{(f_{z_1}+sz_{z_1}^2)-(f_{z_2}+sz_{z_2}^2)}{sz_{z_1}-sz_{z_2}} \le 2 \cdot (n-sz_y)
$$
不难发现，如果 $\large sz_y$ 单调增，那么对应的直线的斜率单调减，所以按这个排序。

对于两个决策 $\large k_1,k_2$，且前者优于后者，那么一定有
$$
\large \frac{f_{k_1}+sz_{k_1}^2}{sz_{k_1}} \le \frac{f_{k_2}+sz_{k_2}^2}{sz_{k_2}}
$$
斜率是单调减的，决策集合就是一个上凸壳，单调队列维护即可。

 

最后讨论一下复杂度。

并不好表达，因为排序和斜率优化都是与子节点个数成正比的。

考虑每个点子节点个数的平均值 $\large \xi$。

这个字母能表示随机变量，直接拿来用了（

如果树是一条链，那么每个点的子节点和为
$$
\large \sum_{i=1}^{n-1} i = C_n^2 = \frac{n(n-1)}{2}
$$
则
$$
\large \xi_1 = \frac{n-1}{2}
$$
如果树是一朵菊花，那么只有根有 $\large n-1$ 个子节点。
$$
\large \xi_2 = \frac{n-1}{n}
$$
接着考虑普通树。

对于一颗普通树，不能产生贡献的叶子节点一定多于链，少于菊花。减少根的直接子节点的个数（其实就是叶子节点个数）一定会导致子节点个数和增大，反之只会减小。

所以，对于普通树
$$
\large \xi \in (\xi_2,\xi_1)
$$
总复杂度 $\large O(n \xi_n^2 \log_2 \xi_n)$

实际上挺快的。

#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
#define ll long long
const int N=5e5+6;
ll n, dlt=(1ll<<60), sz[N], p[N], q[N], f[N];
ll c, h[N], ver[N<<1], nxt[N<<1];
inline void add(ll x,ll y) { ver[++c]=y, nxt[c]=h[x], h[x]=c; }
inline bool cmp(ll x,ll y) { return sz[x]<sz[y]; }
inline ll squ(ll x) { return x*x; }
inline ll Min(ll x,ll y) { return x<y? x:y; }
inline ll X(ll x) { return sz[x]; }
inline ll Y(ll x) { return f[x]+squ(sz[x]); }
inline double calc(ll x,ll y) {
    if(X(x)!=X(y)) return 1.0*(Y(x)-Y(y))/(X(x)-X(y));
    return Y(x)>Y(y)? 1e9:-1e9;
    // 横坐标相同，那么如果 x 纵坐标更大，则其斜率更优，保留。反之舍去
}
void dp(ll x,ll fr) {
    ll i, y, cnt=0, l=1, r=0;
    sz[x]=1;
    for(i=h[x],y;i;i=nxt[i]) if(ver[i]!=fr) {
        y=ver[i];
        dp(y,x), sz[x]+=sz[y];
    }
    f[x]=squ(sz[x]);
    if(sz[x]==1) return;
    for(i=h[x];i;i=nxt[i]) if(ver[i]!=fr) {
        y=ver[i], p[++cnt]=y;
        f[x]=Min(f[x],f[y]+squ(sz[x]-sz[y]));
    }
    dlt=Min(dlt,f[x]+squ(n-sz[x]));
    sort(p+1,p+cnt+1,cmp), q[++r]=p[1]; // 最小的一定是决策之一
    for(i=2;i<=cnt;++i) {
        y=p[i];
        while(l<r&&calc(q[l+1],q[l])<=2.0*(n-sz[y])) ++l;
        dlt=Min(dlt,f[y]+f[q[l]]+squ(n-sz[q[l]]-sz[y]));
        while(l<r&&calc(y,q[r-1])<=calc(q[r],q[r-1])) --r;
        q[++r]=y;
        // 一定都是小于等于，不然 WA
    }
}
int main() {
    scanf("%lld",&n);
    for(ll i=1,x,y;i<n;++i) scanf("%lld%lld",&x,&y), add(x,y), add(y,x);
    dp(1,0);
    printf("%lld\n",n*(n-1)/2+(n*n-dlt)/2);
}

luogu3349 小星星 题解

发表于 2021-10-16 分类于 题解 阅读次数：

设 $\large f_{i,j}$ 为树中节点 $\large i$ 映射到图中 $\large j$ 的方案数。

边界 $\large f_{x,i}=1$

转移
$$
\large f_{x,i} = \prod_{y \in son(x)} \sum{f_{y,j}} \quad i,j \in G
$$
答案 $\large \sum_{i \in [1,n]} f_{1,i}$

单纯 DP 复杂度是 $\large O(n^3)$ 的（枚举树上一点，图上两点）。

题目给出了两条限制

  1. 每条树边都要在图中出现
  2. 每个编号仅出现 1 次

不难发现，DP 枚举点的过程已经满足了条件 1。

考虑第二个条件。如果有编号出现不止一次，那么一定有编号没有出现，等价于有点没有被选到。我们无法直接求出答案，但不难发现，我们可以构造容斥。

0 个点没有被选的方案数-至少 1 个点没有被选的方案数+至少 2 个节点没有被选的方案数……

所以枚举每个点被选的情况，容斥一下，最后显然就是正确答案了。

总复杂度 $\large O(2^nn^3)$

#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
#define ll long long
const int N=20;
int n, m, s, tot;
int cnt, h[N], ver[N<<1], nxt[N<<1];
int tc, hc[N], vc[N*N], nc[N*N];
ll ans, dlt, f[N][N];
void add(int x,int y) { ver[++cnt]=y, nxt[cnt]=h[x], h[x]=cnt; }
void addc(int x,int y) { vc[++tc]=y, nc[tc]=hc[x], hc[x]=tc; }
void dp(int x,int fr) {
    int i, j, k, y;
    bool fg=0;
    for(i=h[x];i;i=nxt[i]) if(ver[i]!=fr) {
        y=ver[i];
        dp(y,x), fg=1;
    }
    if(!fg) {
        for(i=1;i<=n;++i) if(!((s>>(i-1))&1)) f[x][i]=1;
        // 特判
        return;
    }
    for(i=1;i<=n;++i) if(!((s>>(i-1))&1)) {
        f[x][i]=1;
        for(j=h[x];j;j=nxt[j]) if(ver[j]!=fr) {
            y=ver[j];
            ll o=0;
            for(k=hc[i];k;k=nc[k]) o+=f[y][vc[k]];
            f[x][i]*=o;
        }
    }
}
int main() {
    int i, x, y;
    scanf("%d%d",&n,&m);
    for(i=1;i<=m;++i) scanf("%d%d",&x,&y), addc(x,y), addc(y,x);
    for(i=1;i<n;++i) scanf("%d%d",&x,&y), add(x,y), add(y,x);
    for(s=0;s<(1<<n);++s) {
        tot=dlt=0;
        for(i=0;i<n;++i) if((s>>i)&1) ++tot;
        memset(f,0,sizeof(f));
        dp(1,0);
        for(i=1;i<=n;++i) dlt+=f[1][i];
        if(tot&1) ans-=dlt; else ans+=dlt;
    }
    printf("%lld\n",ans);
}
