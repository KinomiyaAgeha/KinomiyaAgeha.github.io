luogu3267 侦察守卫 题解

发表于 2021-10-16 分类于 题解 阅读次数：

本题难点，在于设计「互补」的状态。

每个守卫覆盖的 $\large d$ 的方向是任意的，不能再像传统树形 DP 那样状态单单表示子树信息。

定义「上方」为向根的方向，「下方」为子树内。

设 $\large f_{x,i}$ 为覆盖 $\large x$ 下方的关键点，并且能够向上方覆盖 $\large i$ 范围内的关键点，所需要的最小代价。

设 $\large g_{x,i}$ 为覆盖 $\large x$ 下方 $\large [i,d]$ 范围内的关键点，所需要的最小代价。

这样， $\large f$ 与 $\large g$ 就能拼凑出正确的答案。

 

边界
$$
\large g_{x,i}=w_x, g_{x,d+1}=\inf \quad i \in [1,d]
$$
考虑转移。
$$
\large g_{x,i} = \min_{y \in son(x)}{ { f_{y,i}+g_{x,i},f_{x,i+1}+g_{y,i+1} } }
$$

因为 $\large [i,d]$ 的代价一定能够覆盖 $\large [i-1,d]$，所以

$$
\large g_{x,i} = \min_{i \in [0,d]} { { g_{x,i+1}} }
$$

接着不难想到
$$
\large f_{x,0}=g_{x,0}
\
\large f_{x,i}=\sum_{y \in son(x)} {f_{y,i-1}} \quad i \in [1,d+1]
$$
向上覆盖 $\large i+1$ 的代价，一定能够覆盖 $\large i$，所以
$$
\large f_{x,i}=\min_{i \in [1,d+1]}{ { f_{x,i-1} } }
$$
答案 $\large f_{1,0}$

复杂度 $O(nd)$

#include<cstdio>
#include<iostream>
using namespace std;
const int N=5e5+6;
int n, m, d, w[N], f[N][20], g[N][20];
int c, h[N], ver[N<<1], nxt[N<<1];
bool v[N];
void add(int x,int y) { ver[++c]=y, nxt[c]=h[x], h[x]=c; }
void dp(int x,int fr) {
    int i, j, y;
    if(v[x]) f[x][0]=g[x][0]=w[x];
    for(i=1;i<=d;++i) g[x][i]=w[x]; g[x][d+1]=(1<<30);
    for(i=h[x];i;i=nxt[i]) if(ver[i]!=fr) {
        y=ver[i], dp(y,x);
        for(j=d;j>=0;--j) g[x][j]=min(g[x][j]+f[y][j],g[y][j+1]+f[x][j+1]);
        for(j=d;j>=0;--j) g[x][j]=min(g[x][j],g[x][j+1]);
        f[x][0]=g[x][0];
        for(j=1;j<=d+1;++j) f[x][j]+=f[y][j-1];
        for(j=1;j<=d+1;++j) f[x][j]=min(f[x][j],f[x][j-1]);
        // 注意转移的循环顺序
    }
}
int main() {
    int i, x, y;
    scanf("%d%d",&n,&d);
    for(i=1;i<=n;++i) scanf("%d",&w[i]);
    scanf("%d",&m);
    for(i=1;i<=m;++i) scanf("%d",&x), v[x]=1;
    for(i=1;i<n;++i) scanf("%d%d",&x,&y), add(x,y), add(y,x);
    dp(1,0);
    printf("%d\n",f[1][0]);
}

luogu3225 矿场搭建 题解

发表于 2021-10-16 分类于 题解 阅读次数：

本题思路还是比较清晰的。

显然的，求出 v-DCC 并缩点，然后判断方案数。

在本题中，可以只用 Tarjan 算法求出割点。标记割点，从其他的点依次 DFS，统计每颗搜索树上割点的数量（不重复统计）。

若图中没有割点，那么图中任选两个节点都能满足条件，答案 $\large (2,C_n^2)$

若搜索树上割点为 1，则由于树上两点之间有且仅有一条简单路径，所以这颗搜索树中一定至少选一个点，累加答案，累乘方案数。

若割点多于 1，那么对答案是没有贡献的。

因为无论哪一个点被破坏，该搜索树都会分裂成为上述两种情况。

这种做法实现细节较多。

姑且算是 $\large O(n^2)$？

#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
#define R register
#define SET(x,y) memset(x,y,sizeof(x))
const int N=505;
int t, n, m, num, cnt, fg, ans1, dfn[N], low[N], v[N], w[N];
int c, h[N], ver[N<<1], nxt[N<<1];
long long ans2;
void add(int x,int y) { ver[++c]=y, nxt[c]=h[x], h[x]=c; }
void tarjan(int x) {
    R int i, y;
    dfn[x]=low[x]=++num, v[x]=1;
    for(i=h[x];i;i=nxt[i]) {
        y=ver[i];
        // v数组在一定程度上起到了dfn数组的作用，可以少清空一个数组
        if(!v[y]) {
            tarjan(y);
            low[x]=min(low[x],low[y]);
            if(dfn[x]<=low[y]) ++v[x];
        } else low[x]=min(low[x],dfn[y]);
    }
    if((x==1&&v[x]>2)||(x>1&&v[x]>1)) v[x]=2;
    // 1为普通点，2为割点。注意v[x]在上面已经初始化为1
}
void dfs(int x,int z) {
    R int i, y;
    v[x]=114514, ++cnt;
    // 放置重复搜索，统计节点数
    for(i=h[x];i;i=nxt[i]) {
        y=ver[i];
        if(v[y]==1) dfs(y,z);
        else if(v[y]==2&&w[y]!=z) ++fg, w[y]=z;
        // 防止搜索成环
    }
}
void sol() {
    SET(v,0), SET(w,0), SET(h,0);
    c=ans1=n=0, ans2=1ll;
    R int i, x, y;
    for(i=1;i<=m;++i) {
        scanf("%d%d",&x,&y);
        add(x,y), add(y,x);
        n=max(n,max(x,y));
    }
    tarjan(1);
    // luogu给出的数据是联通的
    for(i=1,num=0;i<=n;++i) if(v[i]==1) {
        fg=cnt=0, ++num;
        dfs(i,num);
        if(fg==1)  ++ans1, ans2*=cnt;
        // 从非割点搜索
    }
    if(!ans1) ans1=2, ans2=n*(n-1)/2;
    printf("Case %d: %d %lld\n",++t,ans1,ans2);
}
int main() { while(scanf("%d",&m)&&m) sol(); }

luogu3574 FarmCraft 题解

发表于 2021-10-16 分类于 题解 阅读次数：

树形 DP。

设 $\large f_x$ 为完成以 $\large x$ 为根的配送的最短时间。

引入 $\large g_x$ 作为遍历以 $\large x$ 为根的子树并回到 $\large x$，所需要的时间。

首先考虑 $\large g$ 的求法，因为每遍历完一棵子树都要回到 $\large x$，所以有
$$
\large g_x = \sum_{y \in son(x)} g_y+2
$$
而 $\large g_x$ 对求出 $\large f_x$ 有用，当且仅当 $\large g_x$ 处于 $\large f_x$ 的前一棵子树的阶段。

具体地说，设 $\large x$ 子节点集合为 $\large son(x)$，则
$$
\large y \in son(x) \quad g_x \text{ 处于 $y-1$ 阶段，} f_x \text{ 处于 $y$ 阶段}
$$
转移为
$$
\large f_x = \max{ { c_x,f_y+g_x+1 } } \quad y \in son(x)
$$
$\large +1$ 是因为 $\large g_x$ 表示的状态已经回到了 $\large x$，但还没有到 $\large f_y$ 表示的 $\large y$，所以要加上一条路的时间。

而取最大值，则是配置电脑可以在配送的同时进行，如果能处理完时间更长的状态，那么时间更短的状态一定被处理完了。

特别的，当 $\large x=1$ 时，不能先在比特安萨尔处安装，所以
$$
\large f_1 = \max{ { c_1+g_1,f_y+g_1+1 }} \quad y \in son(1)
$$
答案
$$
\large \max{ { f_1,c_1+g_1 } }
$$
如何实现有顺序地遍历子树呢？贪心。

用邻项交换证明，此处省略。

按照
$$
\large g_x-f_x
$$
递减排序。

理论复杂度 $\large O(nk \log_2 k)$，$\large k$ 是 $\large son(x)$ 的平均值。

实际比 $\large O(n^2 \log_2 n)$ 低很多，可以 AC 本题。

#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
#define R register
const int N=5e5+5;
int n, k, c[N], f[N], g[N], u[N];
int cnt, h[N], ver[N<<1], nxt[N<<1];
void add(int x,int y) { ver[++cnt]=y, nxt[cnt]=h[x], h[x]=cnt; }
bool cmp(int x,int y) { return g[x]-f[x]<g[y]-f[y]; }
void dp(int x,int fr) {
    R int i, y, k=0;
    if(x!=1) f[x]=c[x];
    for(i=h[x];i;i=nxt[i]) if(ver[i]!=fr) dp(ver[i],x);
    for(i=h[x];i;i=nxt[i]) if(ver[i]!=fr) u[++k]=ver[i];
    sort(u+1,u+k+1,cmp);
    for(i=1;i<=k;++i) {
        y=u[i];
        f[x]=max(f[x],f[y]+g[x]+1), g[x]+=g[y]+2;
    }
}
int main() {
    R int i, x, y;
    scanf("%d",&n);
    for(i=1;i<=n;++i) scanf("%d",&c[i]);
    for(i=1;i<n;++i) scanf("%d%d",&x,&y), add(x,y), add(y,x);
    dp(1,0);
    printf("%d\n",max(f[1],g[1]+c[1]));
}

luogu1967 货车运输 题解

发表于 2021-10-10 更新于 2021-10-16 分类于 题解 阅读次数：

题目要求不超过限重，不难想到因该最大化每条路的限重。所以在原图上求出最大生成树。

那么对于点 $(x,y)$，如果在并查集中 $x$ 与 $y$ 不在同一个集合，则 $x$ 不能到达 $y$

接下来就是每辆车最多运送的货物，不难想到最多运送的货物就是 $\large (x \rightarrow y)$ 路径上权值最小的边。

如果用朴素的算法去求最小的边权，那么复杂度会上天，$\large O(n)$。

联系我们对想上标记法的倍增优化，不难想到对求路径上最小的边权进行倍增优化。

设 $\large d(x,k)$ 为节点 $\large x$ 到它的 $\large 2^k$ 辈祖先这条路径上最小的边权。

接着不难想到转移
$$
\large d(x,k)=\min{ { d(x,k-1),d(f(x,k-1),k-1) } } \quad k \le \log_2n
$$
可以在求 $\large f$ 数组的同时求出。

所以，在求 LCA 的过程中不断维护路径上最小的 $\large d(x,k)$。

这题毒瘤数据，给出的图不一定联通。

#include<bits/stdc++.h>
using namespace std;
#define R register
const int N=1e5+10, M=5e5+5;
int n, m, q, t, o;
int fr[N], f[N][20], d[N][20], dep[N];
int cnt, h[N], ver[N<<1], nxt[N<<1], g[N<<1];
struct pt { int u, v, w; } a[M];
void add(int x,int y,int z) { ver[++cnt]=y, g[cnt]=z, nxt[cnt]=h[x], h[x]=cnt; }
int get(int x) { return x==fr[x]? x:fr[x]=get(fr[x]); }
bool cmp(pt a,pt b) { return a.w>b.w; }
void kruskal() {
    sort(a+1,a+m+1,cmp);
    R int i, x, y;
    for(i=1;i<=m;++i) {
        x=get(a[i].u), y=get(a[i].v);
        if(x!=y) {
            fr[x]=y;
            add(a[i].u,a[i].v,a[i].w), add(a[i].v,a[i].u,a[i].w);
        }
    }
}
void dfs(int x,int pre) {
    R int i, y;
    dep[x]=dep[pre]+1;
    for(i=1;i<=17;++i) {
        f[x][i]=f[f[x][i-1]][i-1];
        d[x][i]=min(d[x][i-1],d[f[x][i-1]][i-1]);
    }
    for(i=h[x];i;i=nxt[i]) if(ver[i]!=pre) {
        y=ver[i];
        f[y][0]=x, d[y][0]=g[i], dfs(y,x);
    }
}
int lca(int x,int y) {
    R int i, res=1<<30;
    if(get(x)!=get(y)) return -1;
    if(dep[x]<dep[y]) swap(x,y);
    for(i=17;i>=0;--i) {
        if(dep[f[x][i]]>=dep[y]) res=min(res,d[x][i]), x=f[x][i];
        if(x==y) return res;
    }
    for(i=17;i>=0;--i) if(f[x][i]!=f[y][i]) {
        res=min(min(res,d[x][i]),d[y][i]);
        x=f[x][i], y=f[y][i];
    }
    return min(min(res,d[x][0]),d[y][0]);
}
int main() {
    R int i, x, y;
    memset(d,0x3f,sizeof(d));
    scanf("%d%d",&n,&m);
    for(i=1;i<=n;++i) fr[i]=i;
    for(i=1;i<=m;++i) scanf("%d%d%d",&a[i].u,&a[i].v,&a[i].w);
    kruskal();
    for(i=1;i<=n;++i) if(!dep[i]) dfs(i,0);
    scanf("%d",&q);
    while(q--) {
        scanf("%d%d",&x,&y);
        printf("%d\n",lca(x,y));
    }
}

CF372C Watching Fireworks is Fun 题解

发表于 2021-10-10 分类于 题解 阅读次数：

题外话

前天英语考试，作文是

假如你是李华，你的美国朋友 Jack 对中国传统文化很感兴趣，他写信希望你能够告诉他有关春节的事情，请你写一封回信给他。开头和结尾已经给出，不计入总词数……

然后因为做过这道题，我就记住了 Firework 这个词，在这篇作文中竟然用上了（

solution

设 $\large f_{i,j}$ 为第 $\large i$ 个烟花 $\large blooms$ 时，在第 $\large j$ 个位置所能得到的最大开心值。

边界 $\large f_{0,i}=0 \quad i \in [1,n]$

转移是显然的（好像这类题的朴素转移都挺显然的）
$$
\large f_{i,j} = \min_{1 \le k \le n} { { f_{i-1,k} + b_i - \left| a_i-j \right| } }
$$
然后得到了一个喜人的复杂度 $\large O(mn^2)$

考虑优化，期望复杂度 $\large O(mn)$。

首先是 $\large n \le 1.5 \times 10^5,m \le 300$，空间不能承受，并且阶段 $\large i$ 只与阶段 $\large i-1$ 有关，滚动数组优化即可。

优化后只有两维，设当前维度为 $\large w$，另一维为 $\large w \text{ xor } 1$

不难发现转移的瓶颈在于枚举 $\large k$，且状态符合 1D/1D 模型，考虑单调队列优化。

 

设 $\large i-1$ 阶段，位置在 $\large k$。对于 $\large (i,j)$，能够向右移动到为区间为
$$
\large [k, \min \big( j+ d \cdot (t_i-t_{i-1}) \big) ]
$$
枚举每个合法的 $\large k$，在单调队列中维护 $\large f_{w \text{ xor } 1,j}$ 单调减。

能够向左到达的区间为
$$
\large [ \max \big( 1,j- d \cdot (t_i-t_{i-1}) \big), k ]
$$
排除掉队首小于左边界的决策即可。

 

设队首为 $\large k$

最終の轉移！
$$
\large f_{w,j} = f_{w \text{ xor } 1,k}+b_i- \left| a_i-j \right|
$$
虽然有枚举合法去年内的决策，但无伤大雅，可以看作常数。

复杂度 $\large O(mn)$。

#include<cstdio>
#include<iostream>
using namespace std;
#define R register
#define ll long long
const int N=150005, M=305;
int n, m, d, l, r, w=1, q[N];
ll a[M], b[M], t[M], f[2][N];
void sol() {
    R int i, j, k, o, u;
    for(int i=1;i<=n;++i) f[0][i]=0;
    for(int i=1;i<=m;++i) {
        l=k=1, r=0;
        for(int j=1;j<=n;++j) {
            o=min(1ll*n,j+d*(t[i]-t[i-1]));
            u=max(1ll,j-d*(t[i]-t[i-1]));
            for(;k<=o;++k) {
                while(l<=r&&f[w^1][q[r]]<=f[w^1][k]) --r;
                q[++r]=k;
            }
            while(l<=r&&q[l]<u) ++l;
            f[w][j]=f[w^1][q[l]]-abs(a[i]-j)+b[i];
        }
        w^=1;   
    }
}
signed main() {
    R int i;
    R ll ans=-(1ll<<60);
    scanf("%d%d%d",&n,&m,&d);
    for(i=1;i<=m;++i) scanf("%lld%lld%lld",&a[i],&b[i],&t[i]);
    sol();
    for(int i=1;i<=n;++i) ans=max(ans,f[w^1][i]);
    printf("%lld\n",ans);
}

POJ3017 Cut the Sequence 题解

发表于 2021-10-10 分类于 题解 阅读次数：

设 $\large f_i$ 为把前 $\large i$ 个数合法分成若干段，每段最大值之和的最小值。
$$
\large f_i = \min_{0 \le j < i , and , \sum_{k=j+1}^i A_k \le M} { \left { f_j + \max_{j+1 \le k \le i } { A_k } \right } }
$$
显然是 $\large O(n^2)$ 的，考虑优化。

若 $\large j$ 可能为最优决策，则它一定满足
$$
\large \sum_{k=j+1}^i A_k \le M \text{ 且 } \sum_{k=j}^i A_k > M
\
\large \text{即 $\large j$ 是满足左式的最小决策}
$$
或
$$
\large A_j = \max_{j \le k \le i}{ { A_k } }
\
\large \text{ 即 $A_j$ 是 $\large A$ 在 $\large [j,i]$ 上的最大值 }
$$
 

证明：

反证法。 若最优决策 $\large j$ 不满足上述两个条件，那么
$$
\large \max_{j \le k \le i}{ { A_k } } = \max_{j+1 \le k \le i}{ { A_k } }
$$
因为 $\large f$ 有单调性，所以 $\large f_{j-1} \le f_j$

即
$$
\large f_{j-1}+\max_{j \le k \le i}{ { A_k } } \le f_j + \max_{j+1 \le k \le i}{ { A_k } }
$$
与 $\large j$ 是最优决策矛盾，固假设不成立，命题得证。

 

对于第一个条件，可以直接预处理 $\large \sum_{k=j+1}^i A_k \le M$ 的最小的 $\large j$，记为 $\large c_i$。

对于第二个条件，一个决策 $\large j_1$ 无用，当且仅当存在 $\large j_1 \le j_2 \text{ 且 } A_{j_1} \le A_{j_2}$，用单调队列排除 $\large j \le c_i$ 的不合法决策，维护 $\large A_j$ 单调减即可。

但是 $\large \max_{j+1 \le k \le i}{ { A_k } }$ 这个东西没有单调性，队头仅仅是最优决策的一种可能。

考虑用平衡树维护，维护 $\large f_j + \max_{j+1 \le k \le i}{ { A_k } }$ 的较小值。

仔细观察，不难发现，由于有第二个条件的限制，所以对于队列中每个决策 $\large j$
$$
\large \max_{j+1 \le k \le i}{ { A_k } } = A_{q_{j+1}}
$$
所以维护 $\large f_j + A_{j+1} $ 即可。

复杂度 $\large O(n\log_2n)$

#include<cstdio>
#include<iostream>
#include<set>
using namespace std;
const int N=1e5+6;
#define ll long long
int n, a[N], c[N], q[N];
ll m, sum, f[N];
multiset<ll> s;
int main() {
    scanf("%d%lld",&n,&m);
    for(int i=1;i<=n;++i) scanf("%d",&a[i]);
    for(int i=1,j=0;i<=n;++i) {
        for(sum+=a[i];sum>m;++j) sum-=a[j+1];
        c[i]=j;
    }
    int l=1, r=0;
    for(int i=1;i<=n;++i) {
        while(l<=r&&q[l]<=c[i]) s.erase(f[q[l]]+a[q[l+1]]), ++l;
        while(l<=r&&a[q[r]]<=a[i]) s.erase(f[q[r-1]]+a[q[r]]), --r;
        if(l<=r) s.insert(f[q[r]]+a[i]);
        q[++r]=i, f[i]=f[c[i]]+a[q[l]];
        if(s.size()) f[i]=min(f[i],*s.begin());
    }

luogu2680 运输计划 题解

发表于 2021-10-04 更新于 2021-10-06 分类于 题解 阅读次数：

最小化完成所有任务的时间，考虑二分答案。

用 $lca$ 算法预处理每个计划的距离，设第 $i$ 个计划的距离为 $\large W_i$

如何判断完成任务的时间 $t$ 是否可行呢？

不难想到，有以下两种情况。

  1. $\large \max_{1 \le i \le m}{ { W_i }} \le t$。
  2. 将一条边权 为 $dlt$ 的边改为 0（虫洞）后，$\large \max_{1 \le i \le m}{ { W_i - dlt }} \le t$。

第一种情况很容易判断。

对于第二种情况，不难发现：将一条边权为 $len$ 的边改为 0 后能使所有距离大于 $t$ 的计划距离都变成不大于$t$，当且仅当这些计划交于此边，且 $\large len \ge \max_{1 \le i \le m}{ { W_i - t }}$

这是显然的，证明略。

明确这个之后，问题仅仅在于，如何快速统计每条边被计划经过的次数和权。

边权不难求，用倍增 $lca$ 的 $f$ 与 $dis$ 数组即可。设 $(x \rightarrow f(x,0))$ 边权为 $d_x$，到根的距离为 $dis_x$

则 $\large d_x=dis_x-dis(f(x,0)) $

考虑统计经过次数。

如果朴素地去统计，复杂度是 $\large O(n)$ 的。可以用树上差分。

用一个计数数组 $\large K$ 记录差分，计划 $\large (x \rightarrow y)$，令
$$
\large K(x)+1,K(y+1),K(z)-2 \quad z=lca(x,y)
$$
然而这个不像一般树上差分一样，统计子树信息。而且上述做法是边权差分，需要转化成点。

考虑每个计划路径是一条链，所以只需要统计每条链的信息就行了。

我们记录搜索树中每次访问到的点，即为 $\large idf$。因为搜索树是一个 DAG，所以这个东西倒序就是逆拓扑序。

按照这个顺序令 $\large K(f(idf(x),0))+K(idf(x))$ 就能自底向上统计出每个点的信息。

然后判断上述两种情况就行了。

#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
const int N=3e5+10;
int n, m, cnt, d[N], dis[N], f[N][18], k[N], idf[N];
int c, h[N], ver[N<<1], nxt[N<<1], w[N<<1];
struct E { int x, y, z, w; } g[N<<1];
void add(int x,int y,int z) { ver[++c]=y, w[c]=z, nxt[c]=h[x], h[x]=c; }
int r_() {
    int a=0; char c=getchar();
    for(;!isdigit(c);c=getchar());
    for(;isdigit(c);a=(a<<1)+(a<<3)+(c^48),c=getchar());
    return a;
}
void dfs(int x,int fr) {
    int i, y, z;
    idf[++cnt]=x, d[x]=d[fr]+1, f[x][0]=fr;
    for(i=1;(1<<i)<d[x];++i) f[x][i]=f[f[x][i-1]][i-1];
    for(i=h[x];i;i=nxt[i]) if(ver[i]!=fr) {
        y=ver[i], z=w[i];
        dis[y]=dis[x]+z, dfs(y,x);
    }
}
int lca(int x,int y) {
    int i, j;
    if(d[x]<d[y]) swap(x,y);
    for(i=0,j=d[x]-d[y];j;++i,j>>=1) if(j&1) x=f[x][i];
    if(x==y) return x;
    for(i=17;i>=0;--i) if(f[x][i]!=f[y][i]) x=f[x][i], y=f[y][i];
    return f[x][0];
}
bool C(int x) {
    int i, tot=0, ans=0;
    memset(k,0,sizeof(k));
    for(i=1;i<=m;++i) if(g[i].w>x) {
        ++k[g[i].x], ++k[g[i].y], k[g[i].z]-=2;
        ans=max(ans,g[i].w-x), ++tot;
    }
    if(!tot) return 1;
    for(i=n;i;--i) k[f[idf[i]][0]]+=k[idf[i]];
    for(i=2;i<=n;++i) if(k[i]==tot&&dis[i]-dis[f[i][0]]>=ans) return 1;
    return 0;
}
int main() {
    int i, x, y, z, l=0, r=0, mid;
    n=r_(), m=r_();
    for(i=1;i<n;++i) {
        x=r_(), y=r_(), z=r_();
        r+=z, add(x,y,z), add(y,x,z);
    }
    dfs(1,0);
    for(i=1;i<=m;++i) {
        x=r_(), y=r_(), z=lca(x,y);
        g[i].x=x, g[i].y=y, g[i].z=z;
        g[i].w=dis[x]+dis[y]-(dis[z]<<1);
    }
    while(l<r) {
        mid=l+r>>1;
        if(C(mid)) r=mid; else l=mid+1;
    }
    printf("%d\n",l);
}

luogu2446 大陆争霸 题解

发表于 2021-10-02 更新于 2021-10-07 分类于 题解 阅读次数：

每个点都必须在到到达所有保护它的点后才能进入，我们用一种类似拓扑排序的方式求解。

设 p(x)p(x)p(x) 为能够到达节点 xxx 最早的时间，q(x)q(x)q(x) 为能够进入节点 xxx 最早的时间，d(x)d(x)d(x) 为摧毁 xxx 最早的时间。

设 ind(x)ind(x)ind(x) 为保护节点 xxx 的点的个数。

显然 p(x)p(x)p(x) 可以直接用最短路算法求出。

设 (x→y)(x \rightarrow y)(x→y)，则

q(y)=max{d(x)}q(y)= \max{\{d(x)\}} q(y)=max{d(x)}

xxx 被摧毁后自然能够到达 yyy。

当保护节点 xxx 的点处理完之后，就能进行 d(x)d(x)d(x) 的转移。

因为有无限多的机器人，所以节点 xxx 能够到达的那一刻就能够被摧毁。

d(x)=max{p(x),q(x)}d(x)= \max{\{ p(x),q(x) \}} d(x)=max{p(x),q(x)}

具体细节看代码。

#include<bits/stdc++.h>
using namespace std;
#define R register
#define PII pair<int,int>
#define mp make_pair
const int N=3e3+10, M=1e6+10;
int n, m, d[N], p[N], q[N], ind[N];
int cnt, h[N], ver[M<<1], nxt[M<<1], w[M<<1];
int tc, hc[N], vc[M<<1], nc[M<<1];
bool v[N];
void add(int x,int y,int z) { ver[++cnt]=y, w[cnt]=z, nxt[cnt]=h[x], h[x]=cnt; }
void addc(int x,int y) { vc[++tc]=y, nc[tc]=hc[x], hc[x]=tc; }
void dijk() {
	R int i, x, y, z;
	R priority_queue<PII > pq;
	memset(d,0x3f,sizeof(d)), memset(p,0x3f,sizeof(p));
	d[1]=p[1]=q[1]=0, pq.push(mp(0,1));
	while(pq.size()) {
		x=pq.top().second, pq.pop();
		if(v[x]) continue;
		v[x]=1;
		for(i=h[x];i;i=nxt[i]) {
			y=ver[i], z=w[i];
			if(p[y]>d[x]+z) {
				p[y]=d[x]+z;
				if(!ind[y]) d[y]=max(p[y],q[y]), pq.push(mp(-d[y],y));
			}
		}
		for(i=hc[x];i;i=nc[i]) {
			y=vc[i], q[y]=max(q[y],d[x]);
			if(--ind[y]==0) {
				d[y]=max(p[y],q[y]);
				pq.push(mp(-d[y],y));
			}
		}
	}
}
int main() {
	R int i, j, x, y, z;
	scanf("%d%d",&n,&m);
	for(i=1;i<=m;++i) scanf("%d%d%d",&x,&y,&z), add(x,y,z);
	for(i=1;i<=n;++i) {
		scanf("%d",&x);
		while(x--) scanf("%d",&y), ++ind[i], addc(y,i);
	}
	dijk();
	printf("%d\n",d[n]);
}

luogu3953 逛公园 题解

发表于 2021-10-02 更新于 2021-10-07 分类于 题解 阅读次数：

首先跑最短路，本题并不卡那个死掉的算法。

求出 1 号节点到每个点的最短路 $\large d$。

然后考虑计数。

直觉 DP。

设计一个类似于分层图的状态
$$
\large f(x,k) \text{ , 为 1 号节点到 x 号节点，距离为 $d(x)+k$ 的方案数}
$$
考虑转移，设 $\large f(y,k_2)$ 能转移到 $\large f(x,k_1)$。

设 $\large (x \rightarrow y)$ 边权为 $\large z$，则有
$$
\large d(y)+z+k_2=d(x)+k_1 \
\large k_2=d(x)-d(y)-z+k_1
$$
设 $\large V$ 为点集。

转移为
$$
\large f(x,k)= \sum_{y \in V} {f(y,d(x)-d(y)+k-z)}
$$
转移顺序显然是一个显式的拓扑序。

不难发现这样的状态不容易被线性推出，直觉告诉我们要用记忆化搜索。

边界 $\large f(1,0)=1$

答案
$$
\large \sum_{0 \le i \le K} f(n,i)
$$
所以我们建一张反图，因为记忆化搜索是自底向上的。

 

考虑无解的情况。

题目描述中告诉我们有 0 边。

我们发现，有无穷多条合法路径，当且仅当有边权为 0 的环。

又因为没有负边权，所以 0 环一定由若干 0 边构成。

如何找出 0 环呢？

观察我们的转移，不难想到，在 0 边上，$\large k=d(x)-d(y)+k-z$，即会递归到 $\large f(y,k)$

如果有 0 环，那么一定又会再次访问到 $\large f(x,k)$。

所以对于每个 $\large f(x,k)$，用 $\large v(x,k)$ 来记录访问情况。每访问到一个状态，将其记为 1，访问完之后就改回 0。如果访问到 1 的状态，就证明有 0 环，无解。

#include<bits/stdc++.h>
using namespace std;
#define R register
#define PII pair<int,int>
#define mp make_pair
#define SET(x,y) memset(x,y,sizeof(x))
const int N=1e5+10;
int t, n, m, K, P, QwQ, f[N][66], d[N];
int cnt, h[N], ver[N<<1], nxt[N<<1], w[N<<1];
int cnt2, h2[N], ver2[N<<1], nxt2[N<<1], w2[N<<1];
bool v[N], vis[N][66];
int r_() {
    R int a=0; R char c=getchar();
    for(;!isdigit(c);c=getchar());
    for(;isdigit(c);a=(a<<1)+(a<<3)+(c^48),c=getchar());
    return a;
}
void add(int x,int y,int z) { ver[++cnt]=y, w[cnt]=z, nxt[cnt]=h[x], h[x]=cnt; }
void add2(int x,int y,int z) { ver2[++cnt2]=y, w2[cnt2]=z, nxt2[cnt2]=h2[x], h2[x]=cnt2; }
void dijk() {
	R int i, x, y, z;
	R priority_queue<PII > q;
	SET(d,0x3f);
	d[1]=0, q.push(mp(0,1));
	while(q.size()) {
		x=q.top().second, q.pop();
		if(v[x]) continue;
		v[x]=1;
		for(i=h[x];i;i=nxt[i]) {
			y=ver[i], z=w[i];
			if(d[y]>d[x]+z) d[y]=d[x]+z, q.push(mp(-d[y],y));
		}
	}
}
int dp(int x,int k) {	
	if(k<0) return 0;
	if(vis[x][k]) { QwQ=1; return 0; }
	if(f[x][k]) return f[x][k];
	R int i, y, z, res=0;
	vis[x][k]=1;
	for(i=h2[x];i;i=nxt2[i]) {
		y=ver2[i], z=w2[i];
		res=(res+dp(y,d[x]-d[y]+k-z))%P;
		if(QwQ) return 0;
	}
	vis[x][k]=0;
	return f[x][k]=res;
}
int main() {
	R int i, x, y, z;
	for(t=r_();t--;) {
		n=r_(), m=r_(), K=r_(), P=r_();

		f[1][0]=1, z=0;

		if(QwQ) puts("-1"); else printf("%d\n",z);
		QwQ=cnt=cnt2=0;
        SET(v,0), SET(vis,0), SET(f,0), SET(h,0), SET(h2,0);

luogu1600 天天爱跑步 题解

发表于 2021-10-02 更新于 2021-10-07 分类于 题解 阅读次数：

经典树上差分。

考虑能观察到玩家的条件，不难发现，对于每个观察员 $x$，能观察到玩家 $i$，当且仅当满足

  1. $\large d(s_i)-d(x)=w_x$
  2. $\large d(s_i)+d(x)-2 \times d(lca(s_i,t_i))=w_x$

上述两式可化为
$$
\large d(s_i)=w_x+d(x) \
\large d(s_i)-2 \times d(lca(s_i,t_i))=w_x-d(x)
$$
观察员可以看作点。

对于一条 $\large (s_i \rightarrow t_i)$ 的路径，由于玩家会经过路径上的每一个点，等号左边是个定值，所以问题可以转化为

对于每条路径，将路径上每一个点都加上两个权值为等号左边的物品。询问每个点权值等于等号右边的物品的个数，两种物品分别计数。

 

显然是树上差分。

对于每个玩家 $\large (x,y)$，求出 $\large z=lca(x,y)$，令
$$
\large x+d(x), , father(z)-d(x) \
\large y+d(x)+2 \times d(z), , z-(d(x)+2 \times d(z))
$$
然后考虑计数。

用 $\large a1,a2$ 记录两种加法操作，用 $\large b1,b2$ 记录两种减法操作。

用值域数组 $\large c1,c2$ ，分别记录两种操作的物品个数。

则要对应着差分来计数，对于每个点的操作，令对应位置加 1 或减 1。

设计数前 $\large p=c1(d(x)+w_x), , q=c2(w_x-d(x))$。

计数后集合得到答案
$$
\large ans(x)=c1(d(x)+w_x)-p+c2(w_x-d(x))-q
$$
注意 $\large w_x-d(x)$ 可能为负，平移一下下标就好了。

#include<cstdio>
#include<iostream>
#include<vector>
using namespace std;
#define pb push_back
#define sz size
const int N=3e5+10;
int n, m, d[N], f[N][18], w[N], c1[N<<1], c2[N<<1], ans[N];
bool v[N];
int cnt, h[N], ver[N<<1], nxt[N<<1];
vector<int> a1[N], a2[N], b1[N], b2[N];
void add(int x,int y) { ver[++cnt]=y, nxt[cnt]=h[x], h[x]=cnt; }
void dfs(int x,int fr) {
    int i, y;
    d[x]=d[fr]+1, f[x][0]=fr;
    for(i=1;(1<<i)<d[x];++i) f[x][i]=f[f[x][i-1]][i-1];
    for(i=h[x];i;i=nxt[i])
        if(ver[i]!=fr) y=ver[i], dfs(y,x);
}
int lca(int x,int y) {
    int i, j;
    if(d[x]<d[y]) swap(x,y);
    for(i=0,j=d[x]-d[y];j;++i,j>>=1) if(j&1) x=f[x][i];
    if(x==y) return x;
    for(i=17;i>=0;--i) if(f[x][i]!=f[y][i]) x=f[x][i], y=f[y][i];
    return f[x][0];
}
void kawaii(int x) {
    int i, y, p=c1[w[x]+d[x]], q=c2[w[x]-d[x]+n];
    v[x]=1;
    for(i=h[x];i;i=nxt[i]) if(!v[ver[i]]) y=ver[i], kawaii(y);
    for(i=0;i<a1[x].sz();++i) ++c1[a1[x][i]];
    for(i=0;i<b1[x].sz();++i) --c1[b1[x][i]];
    for(i=0;i<a2[x].sz();++i) ++c2[a2[x][i]+n];
    for(i=0;i<b2[x].sz();++i) --c2[b2[x][i]+n];
    ans[x]+=c1[d[x]+w[x]]-p+c2[w[x]-d[x]+n]-q;
}
signed main() {
    int i, x, y, z;
    scanf("%d%d",&n,&m);
    for(i=1;i<n;++i) {
        scanf("%d%d",&x,&y);
        add(x,y), add(y,x);
    }
    for(i=1;i<=n;++i) scanf("%d",&w[i]);
    dfs(1,0);
    while(m--) {
        scanf("%d%d",&x,&y), z=lca(x,y);
        a1[x].pb(d[x]), b1[f[z][0]].pb(d[x]);
        a2[y].pb(d[x]-2*d[z]), b2[z].pb(d[x]-2*d[z]);
    }
    kawaii(1);
    for(i=1;i<=n;++i) printf("%d ",ans[i]);
}
