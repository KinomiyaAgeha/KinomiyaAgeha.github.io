luogu4381 Island 题解

发表于 2021-10-01 更新于 2021-10-07 分类于 题解 阅读次数：

不难发现，给出的是一个基环树森林。

渡船的使用条件，实际上是：离开一颗基环树后，就不能再回来。

题目要求走过的路最长，不难想到是基环树的直径。

显然基环树的直径有两种可能

  1. 在去掉环后的某棵子树中（若有负边权）
  2. 两端在在去掉环后的两颗子树中 / 把环断开后，树的直径

（但在本题中只有第二种就是了）

最终答案为每颗基环树直径的和。

对于第一种情况，找到环后，以环上的每个节点为根，在它的子树中跑 DP 求最长链就行了。

设以环上节点 $\large x$ 为根，不经过换上节点，能够到达的最远距离为 $d(x)$，两点间距离为 $\large dis(x,y)$。

第二种情况仅仅是：选择两个环上的点 $\large (i,j)$，最大化 $\large d(i)+d(j)+dis(i,j)$。

考虑环形 DP 的处理方案，我们将环断开并复制一倍，用单调队列优化点的枚举。

用前缀和处理两点间的距离，设其为 $S$。

设环为 $\large u$。

在队头维护：满足 $\large d(u_i)+d(u_j)+S(u_j)-S(u_i)$ 的最大的 $\large u_j$。

在队尾维护：$\large d(u_r)-S(u_r)$ 单调减。

环上两点有顺时针和逆时针两个距离，本题无负权，所以其中一个距离一定大于另一个。

若环上有 $\large p$ 个节点，则在队头排除距离小于 $\large i-p$ 的决策。

实现时注意细节。

#include<bits/stdc++.h>
using namespace std;
#define R register
#define ll long long
const int N=1e6+5;
int n, num, p, cnt;
int dfn[N], fr[N], s[N<<1], q[N<<1];
int h[N], nxt[N<<1], ver[N<<1], w[N<<1];
bool v[N];
ll ans, ANS;
ll d[N], st[N<<1];
void add(int x,int y,int z) { ver[++cnt]=y, nxt[cnt]=h[x], w[cnt]=z, h[x]=cnt; }
void init() {
    R int x, y, z;
    scanf("%d",&n);
    for(x=1;x<=n;++x) scanf("%d%d",&y,&z), add(x,y,z), add(y,x,z);
}
void fc(int x,int y,int z) {
    R int i;
    st[1]=z;
    while(x!=y) s[++p]=y, st[p+1]=w[fr[y]], y=ver[fr[y]^1];
    s[++p]=x;
    for(i=1;i<=p;++i) v[s[i]]=1, s[p+i]=s[i], st[p+i]=st[i];
    for(i=1;i<=p<<2;++i) st[i]+=st[i-1];
}
void dfs(int x) {
    R int i, y;
    dfn[x]=++num;
    for(i=h[x];i;i=nxt[i]) {
        y=ver[i];
        if(!dfn[y]) fr[y]=i, dfs(y);
        else if((i^1)!=fr[x]&&dfn[y]>dfn[x]) fc(x,y,w[i]);
    }
}
void DP(int x) {
    R int i, y, z;
    v[x]=1;
    for(i=h[x];i;i=nxt[i]) if(ver[i]!=fr) {
        y=ver[i];
        DP(y);
        ans=max(ans,d[x]+d[y]+z), d[x]=max(d[x],d[y]+z);
    }
}
void korekara(int x) {
    R int i;
    p=ans=0;
    dfs(x);
    for(i=1;i<=p;++i) DP(s[i]);
    R int l=1, r=0;
    for(i=1;i<=p<<1;++i) {
        while(l<=r&&q[l]<=i-p) ++l;
        ans=max(ans,d[s[i]]+d[s[q[l]]]+st[i]-st[q[l]]);
        while(l<=r&&d[s[q[r]]]-st[q[r]]<=d[s[i]]-st[i]) --r;
        q[++r]=i;
    }
    ANS+=ans;
}
int main() {
    cnt=1;
    init();
    for(R int i=1;i<=n;++i) if(!dfn[i]) korekara(i);
    printf("%lld\n",ANS);
}

实际上就是 树网的核 的数据加强版。

原题暴力枚举即可，本题也可以用复杂度为 O(nlogSUM)O(n\log SUM)O(nlogSUM) 的二分答案，这里只讲述单调队列的 O(n)O(n)O(n) 算法。

题意：在树的直径上选择两个距离不超过 s\large ss 的点，最小化「偏心距」。

「偏心距」：树中距离直径最远的节点到直径的距离。

显然，可以用单调队列维护。

设直径为 u\large uu，其节点数为 o\large oo，直径上两点为 (ui,uj)\large (u_i,u_j)(u​i​​,u​j​​)，f(x)\large f(x)f(x) 为从 x\large xx 出发，不经过直径上的其他点，能够到达的最远距离，dis(x,y)\large dis(x,y)dis(x,y) 为 (x,y)\large (x,y)(x,y) 之间的距离。

不难想到此时偏心距为

maxi,j∈[1,s](maxi≤k≤j{f(k)},dis(u1,ui),dis(uj,us))\large \max_{i,j \in [1,s]} {(\max_{i \le k \le j} { \{ f(k) \},dis(u_1,u_i),dis(u_j,u_s))} } ​i,j∈[1,s]​max​​(​i≤k≤j​max​​{f(k)},dis(u​1​​,u​i​​),dis(u​j​​,u​s​​))

仔细观察不难发现，f(k)\large f(k)f(k) 的定义域为 [1,s]\large [1,s][1,s]，而无论如何选择两个点，最终一定取到max1≤k≤sf(k)\large \max_{1\le k \le s}{f(k)}max​1≤k≤s​​f(k)。

所以上式可化为

maxi,j∈[1,s](max1≤k≤s{f(k)},dis(u1,ui),dis(uj,uo))\large \max_{i,j \in [1,s]} {(\max_{1 \le k \le s}{\{ f(k) \}},dis(u_1,u_i),dis(u_j,u_o))} ​i,j∈[1,s]​max​​(​1≤k≤s​max​​{f(k)},dis(u​1​​,u​i​​),dis(u​j​​,u​o​​))

而它是一个定值，设其为 K\large KK。

求出 u\large uu 上节点距离的前缀和数组 SSS。

则

maxi,j∈[1,s](K,max{S(i),S(o)−S(j)})\large \max_{i,j \in [1,s]} { (K,\max \{ S(i),S(o)-S(j)\}}) ​i,j∈[1,s]​max​​(K,max{S(i),S(o)−S(j)})

#include<cstdio>
#include<iostream>
using namespace std;
#define R register
const int N=3e5+10;
int n, s, t, mx, o, d[N], pre[N], f[N], sum[N], a[N], b[N];
int c=1, h[N], ver[N<<1], w[N<<1], nxt[N<<1];
bool v[N];
void add(int x,int y,int z) { ver[++c]=y, w[c]=z, nxt[c]=h[x], h[x]=c; }
void DFS(int x,int fr) {
    R int i, y, z;
    for(i=h[x];i;i=nxt[i]) if(ver[i]!=fr) {
        y=ver[i], z=w[i], pre[y]=i, d[y]=d[x]+z;
        if(d[y]>mx) mx=d[y], o=y;
        DFS(y,x);
    }
}
void dfs(int x) {
    R int i, y, z;
    v[x]=1;
    for(i=h[x];i;i=nxt[i]) if(!v[ver[i]]) {
        y=ver[i], z=w[i];
        dfs(y);
        f[x]=max(f[x],f[y]+z);
    }
}
int main() {
    R int i, j, x, y, z;
    R int l, r, p, q, k=0, ans=1<<30;
    scanf("%d%d",&n,&s);
    for(i=1;i<n;++i) {
        scanf("%d%d%d",&x,&y,&z);
        add(x,y,z), add(y,x,z);
    }
    DFS(1,0), p=o, mx=d[o]=0;
    DFS(p,0), q=o;
    while(q!=p) v[a[++t]=q]=1, b[t+1]=w[pre[q]], q=ver[pre[q]^1];
    v[a[++t]=p]=1;
    for(i=1;i<=t;++i) {
        dfs(a[i]);
        k=max(k,f[a[i]]);
        sum[i]=sum[i-1]+b[i];
    }
    for(l=1,r=1;l<=t;++l) {
        while(r<t&&sum[r+1]-sum[l]<=s) ++r;
        z=max(k,max(sum[l],sum[t]-sum[r]));
        ans=min(ans,z);
    }
    printf("%d\n",ans);
}

luogu3304 直径 题解

发表于 2021-10-01 更新于 2021-10-07 分类于 题解 阅读次数：

两次 DFS 求出树的直径。

显然多条直径必定交于至少一点，且包含它们的中点。

则若舍去他们交点的之外的边，剩下的边即为所求。

设直径左右端点为 $\large l, r$。

在第二次 DFS 时能够求出 $\large l$ 到直径每个节点的距离，所以从 $\large r$ 向 $\large l$ 遍历。

对于直径上的每个点 $i$，分别求出在不经过直径上其他点的情况的，所能达到的最远距离，记作 $\large d$。

设它到直径左端点距离为 $\large ld$，到右端点距离为 $\large rd$。

若 $\large d=rd$，则令 $\large r=i$。

若 $\large d=ld$，则令 $\large l=i$，且只会进行一次，这是因为我们的遍历顺序是从右到左。

#include<cstdio>
#include<iostream>
using namespace std;
#define R register
#define ll long long
const int N=2e5+10;
int n, f[N];
int c, h[N], ver[N<<1], nxt[N<<1], w[N<<1];
ll p, mxd, d[N];
bool v[N];
void add(int x,int y,int z) { ver[++c]=y, w[c]=z, nxt[c]=h[x], h[x]=c; }
void dfs(int x,int fr) {
    R int i, y, z;
    f[x]=fr;
    for(i=h[x];i;i=nxt[i]) if(ver[i]!=fr) {
        y=ver[i], z=w[i], d[y]=d[x]+z;
        if(d[y]>mxd) { mxd=d[y], p=y; }
        dfs(y,x);
    }
}
void kawaii(int x,int fr) {
    R int i, y, z;
    for(i=h[x];i;i=nxt[i]) if(ver[i]!=fr&&!v[ver[i]]) {
        y=ver[i], z=w[i], d[y]=d[x]+z;
        if(d[y]>mxd) mxd=d[y];
        kawaii(y,x);
    }
}
int main() {
    R int i, x, y, z, l, r;
    R bool fg=0;
    scanf("%d",&n);
    for(i=1;i<n;++i) {
        scanf("%d%d%d",&x,&y,&z);
        add(x,y,z), add(y,x,z);
    }
    dfs(1,0), l=p, mxd=d[p]=0;
    dfs(p,0), r=p;
    printf("%lld\n",mxd);
    for(i=r;i;i=f[i]) v[i]=1;
    R int l_=l, r_=r;
    for(i=f[r_];i!=l_;i=f[i]) {
        R int ld=d[i], rd=d[r_]-d[i];
        mxd=d[i]=0;
        kawaii(i,0);
        if(mxd==rd) r=i;
        if(mxd==ld&&!fg) fg=1, l=i;
    }
    for(i=f[r],z=1;i!=l;i=f[i]) ++z;
    printf("%d\n",z);
}

CF1083E The Fair Nut and Rectangles 题解

发表于 2021-09-21 更新于 2021-10-17 分类于 题解 阅读次数：

考虑 DP。

显然 DP 需要一定的顺序。

因为给出的矩形没有包含的关系，所以我们按照每个矩形右上角点的横坐标 $x$ 递增排序，那么纵坐标 $\large y$ 一定是递减排序的，反之亦然。

设 $\large S_i = x_i \times y_i$。

因为每个矩形都有选与不选两种选择，所以状态为

设 $\large f(i)$ 为选择第 $i$ 和矩形获得的最大收益。

初始值为 $\large f(i) = S_i-a_i$

显然，转移为

$$
\large f(i)=\max_{1 \le j < i}{ { f(j)+ S_i - a_i - S_i \cup S_j } }
$$

答案为

$$
\large \max_{1 \le i \le n}{ f(i) }
$$

复杂度为 $\large O(n^2)$。

 

显然，复杂度过高，考虑优化。

由于我们已经将矩形排序，所以

$$
\large \forall S_i \cup S_j \quad x_i \ge x_j,y_i \le y_j
$$

即

$$
\large S_i \cup S_j = x_j \times y_i
$$

所以原方程可化简为
$$
\large f(i) = \max_{ 1 \le j < i } { { f(j)+ x_iy_i - a_i -x_jy_i } }
$$
按照套路去掉 $\large \max$ 函数，移项得
$$
\large f(j)=y_ix_j +f(i) - x_iy_i + a_i
$$
即

$$
\begin{cases}
\large y=f(j) \
\large k=y_i \
\large x=x_j \
\large b= f(i)-x_iy_i+a_i
\end{cases}
$$

对应到坐标系里即为：

每个决策点为 $\large (x_j,f(j))$，其斜率 $\large y_i$ 单调递减，所以要维护一个下凸壳。

由于斜率 $\large y_i$ 单调递减 ，所以合法决策的斜率一定小于 $\large y_i$。

#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
#define ll long long
#define X(y) (w[y].x)
#define Y(x) (f[x])
const int N=1e6+6;
ll n, ans, q[N], f[N];
struct Squ { ll x, y, z; } w[N];
bool operator<(Squ a,Squ b) { return a.y>b.y; }
double calc(ll x,ll y) { return X(x)!=X(y)? 1.0*(Y(x)-Y(y))/(X(x)-X(y)):1e9; }
int main() {
    int i;
    scanf("%lld",&n);
    for(i=1;i<=n;++i) scanf("%lld%lld%lld",&w[i].x,&w[i].y,&w[i].z);
    sort(w+1,w+n+1);
    int l=1, r=1;
    for(i=1;i<=n;++i) {
        f[i]=w[i].x*w[i].y-w[i].z;
        while(l<r&&calc(q[l],q[l+1])>=w[i].y) ++l;
        f[i]=max(f[i],f[q[l]]+(w[i].x-w[q[l]].x)*w[i].y-w[i].z);
        ans=max(ans,f[i]);
        while(l<r&&calc(q[r-1],q[r])<=calc(q[r],i)) --r;
        q[++r]=i;
    }
    printf("%lld\n",ans);
}

luogu5008 锦鲤抄 题解

发表于 2021-09-20 更新于 2021-10-17 分类于 题解 阅读次数：

首先考虑 DAG 的情况。

拓扑序实际上也是「无后效性」的体现，即若存在 $\large (x \rightarrow y)$，$x$ 无论怎样处理都不会影响到 $\large y$。

同理，逆拓扑序也可以理解为无论 $\large y$ 怎样处理，都不会影响 $\large x$。

在本题中，可以选择入度不为 0 但可以没有出边的点.也就是如果删去 $y$ 和它的一条出边，不会使 $\large x$ 受到影响，关于 $\large x$ 的任何操作结果也不会因此变差。

所以只要优先删去权值最大的点就行了，最终会留下一个入度为 0 的点。

我们只要用 Tarjan 算法求出 SCC 并且缩点，就得到了一个 DAG。

 

接下来讨论每个 SCC 内部的选择。

若存在入度为不为 0 的 SCC，设其为 $\large x$。

那么 $\large x$ 内的点可以随便选。

证明：

如果 $\large x$ 的入度不为 0，分以下情况讨论。

对于 $\large x$ 中只有 1 个点的情况，显然成立。

否则，$\large x$ 至少由一个简单环构成，且一定存在入度 $\large \ge 2$ 的点，设其为 $u$。

对于 $\large x$ 中的每个简单环，设其入度最大的点为 $\large u$，则至少可以删去 1 条 $\large u$ 的入边，断开这个简单环.而只要 $\large u$ 的入度不为 0，就不会影响到点的选择。最终可以得到一个多了若干条入边的 DAG。

因为 $\large x$ 入度不为 0，所以这个 DAG 不存在入度为 0 的点，命题得证。

考虑入度为 0 的 SCC，设其为 $\large y$。

仿照上述证明思路，我们仍以 $\large y$ 内每个简单环删去若干边为代价，得到一个 DAG。

不难发现，它是一个普通 DAG，而我们不能选择删去一个入度为 0 的点，设其为 $\large v$。

所以最优解为 $\large v$ 的点权为 $\large y$ 中最小。

更进一步地说，以上关于 SCC 的删边策略，最终得到的都是 SCC 的搜索树，搜索树一定是 DAG。

不同的是前者没有入度为 0 的点，后者有且仅有一个。

 

所以我们按照上述操作之后，把能够选择的点集（决策集合）排个序，取前 $\large k$ 大就可以。

求出 SCC 的顺序即为逆拓扑序。

实现不再赘述。

#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
const int N=5e5+6, M=2e6+6;
int n, m, k, num, tp, dfn[N], low[N], st[N];
int scc, o, c[N], w[N], deg[N], f[N], ans[N];
int cnt, h[N], ver[M], nxt[M];
vector<int> p[N];
void add(int x,int y) { ver[++cnt]=y, nxt[cnt]=h[x], h[x]=cnt; }
void tarjan(int x) {
    int i, y;
    dfn[x]=low[x]=++num, st[++tp]=x;
    for(i=h[x];i;i=nxt[i]) {
        y=ver[i];
        if(!dfn[y]) {
            tarjan(y);
            low[x]=min(low[x],low[y]);
        } else if(!c[y]) low[x]=min(low[x],dfn[y]);
    }
    if(dfn[x]==low[x]) {
        ++scc;
        do {
            y=st[tp--], c[y]=scc, f[scc]=min(f[scc],w[y]);
            p[scc].push_back(y);
        } while(x!=y);
    }
}
void sol() {
    int i, x, y;
    for(x=1;x<=n;++x) for(i=h[x];i;i=nxt[i]) {
        y=ver[i];
        if(c[x]!=c[y]) ++deg[c[y]];
    }
    for(x=1;x<=scc;++x) {
        bool fg=(deg[x]==0);
        for(i=0;i<p[x].size();++i) {
            y=p[x][i];
            if(w[y]==f[x]&&fg) fg=0; else ans[++o]=w[y];
        }
    }
}
int main() {
    int i, x, y, z=0;
    memset(f,0x3f,sizeof(f));
    scanf("%d%d%d",&n,&m,&k);
    for(i=1;i<=n;++i) scanf("%d",&w[i]);
    for(i=1;i<=m;++i) scanf("%d%d",&x,&y), add(x,y);
    for(i=1;i<=n;++i) if(!dfn[i]) tarjan(i);
    sol();
    if(o<=k) for(i=1;i<=o;++i) z+=ans[i];
    else { sort(ans+1,ans+o+1); for(i=o-k+1;i<=o;++i) z+=ans[i]; }
    printf("%d\n",z);
}

luogu2195 HXY造公园 题解

发表于 2021-09-20 更新于 2021-10-07 分类于 题解 阅读次数：

我最喜欢的紫色水题（

给出一个森林，有两种操作。

  1. 询问某个点所在的树的直径
  2. 在两个点所在的两棵树间连一条边，最小化其直径

显然的，对于第一种操作，DP / DFS / BFS 预处理直径，并查集维护每棵树的点就行了。

问题在于高效维护第二种操作。

不难想到，两棵树之间连一条边，相当于合并两个集合。

而最小化新树的直径，显然要在两树直径的中点处连边。

证明：

反证法。若最优点不是直径中点，由于 直径有两个端点 且 树上两点有且仅有一条简单路径，若在非直径中点的 $\large x$ 点连边，则当其接近直径一端时，直径另一端到达它的距离就大于到达直径中点的距离，反之则显然。这与假设不符，故原命题正确。

设连边 $\large (x \rightarrow y)$，$\large l(x)$ 为 $\large x$ 的直径。

则新树的直径只有三种可能。

  1. $\large l(x)$
  2. $\large l(y)$
  3. $\large \lceil \frac{l(x)}{2} \rceil + \lceil \frac{l(y)}{2} \rceil + 1$

合并后求最大值就行了。

#include<cstdio>
#include<iostream>
using namespace std;
const int N=3e5+6;
int n, m, q, ans, f[N], d[N], c[N];
int cnt, h[N], ver[N<<1], nxt[N<<1];
bool v[N];
int r_() {
    int a=0; char c=getchar();
    for(;!isdigit(c);c=getchar());
    for(;isdigit(c);a=(a<<1)+(a<<3)+(c^48),c=getchar());
    return a;
}
void add(int x,int y) { ver[++cnt]=y, nxt[cnt]=h[x], h[x]=cnt; }
int get(int x) { return x==f[x]? x:f[x]=get(f[x]); }
void dp(int x,int fr) {
    int i, y;
    for(i=h[x];i;i=nxt[i]) if(ver[i]!=fr) {
        y=ver[i], dp(y,x);
        ans=max(ans,d[x]+d[y]+1), d[x]=max(d[x],d[y]+1);
    }
}
void miku(int x) { ans=0, dp(x,0), c[x]=ans; }
int main() {
    int i, op, x, y, z;
    n=r_(), m=r_(), q=r_();
    for(i=1;i<=n;++i) f[i]=i;
    for(i=1;i<=m;++i) x=r_(), y=r_(), add(x,y), add(y,x), f[get(x)]=get(y);
    for(i=1;i<=n;++i) if(f[i]==i) miku(i);
    while(q --> 0) {
        op=r_(), x=r_();
        if(op&1) { printf("%d\n",c[get(x)]); continue; }
        y=r_(), x=get(x), y=get(y);
        if(x==y) continue;
        c[x]=max((c[x]+1)/2+(c[y]+1)/2+1,max(c[x],c[y]));
        f[y]=x;
    }
}

luogu4819 杀人游戏 题解

发表于 2021-09-20 更新于 2021-10-17 分类于 题解 阅读次数：

不妨假设平民为白点，杀手为黑点，认识的关系为一条有向边。

求不访问黑点并且知道黑点的最小代价。

若有 $\large n$ 个点，显然每个点为黑的概率为 $\large \frac{1}{n}$。

而每访问一个白点，都能得知与它出边相连的点的颜色。

考虑强连通分量。

不难发现，对于每个强连通分量，只要以概率增加 $\large \frac{1}{n}$ 为代价访问其中一个点，就能得知整个强连通分量的颜色情况。这是因为除了第一个访问的点，其他点的颜色在访问前都已经被确定了，若警察足够聪明，则不会影响总概率。

所以求出强连通分量后进行缩点，我们就得到了一个 DAG。

不难想到，为了减少总访问次数，访问入度不为 0 的 SCC 是不划算的。

简单证明：设缩点后存在 $\large (x \rightarrow y)$ 的边，则访问完 $\large x$ 中所有的点后，必定能知道 $\large y$ 中一个点的信息，所以对于 $\large y$，不需要增加 $\large \frac{1}{n}$ 的访问代价。

所以设缩点后入度为 0 的点的数量为 $s$，则访问到黑点的概率为 $\large \frac{s}{n}$，答案为 $\large \frac{n-s}{n}$。

 

然而缩点后不一定是一棵树的状态，一个强连通分量可能有多条出边。

考虑只含一个点的 SCC，设其为 $\large c$，若其入度为 0，且其能够到达的点的入度均大于 1，那么若最后访问 $\large c$，整张图的情况已经被确定了。若未找到黑点，则 $\large c$ 为黑点。这样可以减少一次访问，且对于任意图，能且仅能减少一次。

所以若存在 $c$，令 $\large s=s-1$ 即可。

#include<cstdio>
#include<iostream>
using namespace std;
const int N=1e6+6;
int n, m, k, num, ans, dfn[N], low[N], st[N];
int scc, c[N], deg[N], sz[N];
int cnt, h[N], ver[3*N], nxt[3*N];
int tc, hc[N], vc[3*N], nc[3*N];
bool v[N];
void add(int x,int y) { ver[++cnt]=y, nxt[cnt]=h[x], h[x]=cnt; }
void adc(int x,int y) { vc[++tc]=y, nc[tc]=hc[x], hc[x]=tc; }
void tarjan(int x) {
    int i, y;
    dfn[x]=low[x]=++num, st[++k]=x;
    for(i=h[x];i;i=nxt[i]) {
        y=ver[i];
        if(!dfn[y]) {
            tarjan(y);
            low[x]=min(low[x],low[y]);
        } else if(!c[y]) low[x]=min(low[x],dfn[y]);
    }
    if(dfn[x]==low[x]) {
        ++scc;
        do y=st[k--], c[y]=scc, ++sz[scc]; while(x!=y);
    }
}
bool f(int x) {
    if(deg[x]||sz[x]!=1) return 0;
    for(int i=hc[x];i;i=nc[i]) if(deg[vc[i]]==1) return 0;
    return 1;
}
int main() {
    int i, x, y;
    scanf("%d%d",&n,&m);
    for(i=1;i<=m;++i) scanf("%d%d",&x,&y), add(x,y);
    for(i=1;i<=n;++i) if(!dfn[i]) tarjan(i);
    for(x=1;x<=n;++x) for(i=h[x];i;i=nxt[i]) {
        y=ver[i];
        if(c[x]!=c[y]&&!v[c[y]]) adc(c[x],c[y]), ++deg[c[y]], v[c[y]]=1;
    }
    for(i=1;i<=scc;++i) if(!deg[i]) ++ans;
    for(i=1;i<=scc;++i) if(f(i)) { --ans; break; }
    printf("%.6lf\n",1.0*(n-ans)/n);
}
