$$
C^{k-1}{k+r-1}-\sum{i=1}^k{C^{k-1}{k+r-n_i-2}}+\sum{1\le i<j \le k}{C^{k-1}{k+r-n_i-n_j-3}}- \cdots + (-1)^k C^{k-1}{k+r-\sum_{i=1}^k{n_i-(k+1)}}
$$
