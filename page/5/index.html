T3

MooFest

两个树状数组维护就行了，细节较多。

#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
#define R register
#define ll long long
const int N=5e4+10;
int n;
ll ans, c[2][N];
struct E { int x, y; } a[N];
bool cmp(E a,E b) { return a.y<b.y; }
void modify(int x,ll y,int z) { for(;x<=5e4;x+=x&-x) c[z][x]+=y; }
ll query(int x,int z) {
    R ll y=0;
    for(;x;x-=x&-x) y+=c[z][x];
    return y;
}
int main() {
    R int i;
    R ll x, y, z=0;
    scanf("%d",&n);
    for(i=1;i<=n;++i) scanf("%d%d",&a[i].y,&a[i].x);
    sort(a+1,a+n+1,cmp);
    for(i=1;i<=n;++i) {
        x=query(a[i].x,0), y=query(a[i].x,1);
        ans+=(x*a[i].x-y)*a[i].y;
        ans+=((z-y)-(i-x-1)*a[i].x)*a[i].y;
        modify(a[i].x,1,0), modify(a[i].x,a[i].x,1);
        z+=a[i].x;
    }
    printf("%lld\n",ans);
}

 

T4

Haircut

树状数组维护逆序对数量的偏移量。

#include<cstdio>
#include<iostream>
using namespace std;
#define R register
#define ll long long
const int N=1e5+10;
int n, a[N];
ll s[N], c[N];
void modify(ll x,ll y) { for(;x<=1e5;x+=x&-x) c[x]+=y; }
ll query(ll x) {
    R ll y=0;
    for(;x;x-=x&-x) y+=c[x];
    return y;
}
int main() {
    R int i;
    R ll dlt, ans=0;
    scanf("%d",&n);
    for(i=1;i<=n;++i) scanf("%d",&a[i]), ++a[i];
    puts("0");
    for(i=1;i<=n;++i) {
        dlt=n-a[i]+2, s[a[i]]+=query(dlt-1);
        modify(dlt,1);
    }
    for(i=2;i<=n;++i) printf("%lld\n",ans+=s[i-1]);
}

抢掠计划

Tarjan 算法求出 SCC，缩点建新图，按拓扑序 DP 最长路即可。

因为建新图的add()函数写错了，我竟然调了一个小时（

#include<cstdio>
#include<iostream>
using namespace std;
#define R register
const int N=5e5+10;
int n, m, s, p, num, k, scc, w[N], sw[N], dfn[N], low[N], stk[N];
int ans, c[N], f[N];
int cnt, h[N], ver[N], nxt[N];
int tc, hc[N], vc[N], nc[N];
bool v[N], b[N], B[N];
void add(int x,int y) { ver[++cnt]=y, nxt[cnt]=h[x], h[x]=cnt; }
void cad(int x,int y) { vc[++tc]=y, nc[tc]=hc[x], hc[x]=tc; }
void tarjan(int x) {
    R int i, y;
    dfn[x]=low[x]=++num, stk[++k]=x, v[x]=1;
    for(i=h[x];i;i=nxt[i]) {
        y=ver[i];
        if(!dfn[y]) {
            tarjan(y);
            low[x]=min(low[x],low[y]);
        } else if(v[y]) low[x]=min(low[x],dfn[y]);
    }
    if(dfn[x]==low[x]) {
        ++scc;
        do {
            y=stk[k--], v[y]=0, c[y]=scc;
            sw[scc]+=w[y];
            if(b[y]) B[scc]=1;
        } while(x!=y);
    }
}
void X() {
    R int i, x, y;
    for(x=1;x<=n;++x) for(i=h[x];i;i=nxt[i]) {
        y=ver[i];
        if(c[x]!=c[y]) cad(c[x],c[y]);
    }
}
void dp() {
    R int i, x, y, z;
    for(x=scc;x;--x) {
        f[x]+=sw[x];
        if(B[x]) ans=max(ans,f[x]);
        for(i=hc[x];i;i=nc[i]) {
            y=vc[i];
            f[y]=max(f[y],f[x]);
        }
    }
}
int main() {
    // freopen("data_\\P3627_3.in","r",stdin);
    R int i, x, y;
    scanf("%d%d",&n,&m);
    for(i=1;i<=m;++i) scanf("%d%d",&x,&y), add(x,y);
    for(i=1;i<=n;++i) scanf("%d",&w[i]);
    scanf("%d%d",&s,&p);
    for(i=1;i<=p;++i) scanf("%d",&x), b[x]=1;
    tarjan(s);
    X(), dp();
    printf("%d\n",ans);
}

Cow Toll Paths

有点玄学的题目。

实际上就是个 Floyd。

但一定要注意点权排序后的顺序问题。

#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
#define R register
#define SET(x,y) memset(x,y,sizeof(x))
#define rep(i,j,k) for(i=(j);i<=(k);++i)
const int N=260;
int n, m, q, d[N][N], dis[N][N], rk[N];
struct E { int z, id; } c[N];
bool cmp(E a,E b) { return a.z!=b.z? a.z<b.z:a.id<b.id; }
int main() {
    R int i, j, k, x, y, z;
    SET(d,0x3f), SET(dis,0x3f);
    scanf("%d%d%d",&n,&m,&q);
    rep(i,1,n) scanf("%d",&c[i].z), c[i].id=i, d[i][i]=0;
    sort(c+1,c+n+1,cmp);
    rep(i,1,n) rk[c[i].id]=i;
    while(m --> 0) {
        scanf("%d%d%d",&x,&y,&z);
        d[rk[x]][rk[y]]=d[rk[y]][rk[x]]=min(d[rk[x]][rk[y]],z);
    }
    rep(k,1,n) rep(i,1,n) rep(j,1,n) if(i!=j) {
        d[i][j]=min(d[i][j],d[i][k]+d[k][j]);
        dis[i][j]=min(dis[i][j],d[i][j]+max(c[k].z,max(c[i].z,c[j].z)));
    }
    while(q --> 0) scanf("%d%d",&x,&y), printf("%d\n",dis[rk[x]][rk[y]]);
}

 

T7

Redundant Paths

初看觉得有些奇怪，仔细思考不难发现就是 Tarjan 算法求出 e-DCC，缩点后统计叶子节点的数量 \(\large ans\)，\(\large (ans+1) \div 2\) 即为答案。

#include<cstdio>
#include<iostream>
using namespace std;
#define R register
const int N=5005, M=10005;
int n, m, num, dcc, ans, dfn[N], low[N], c[N], deg[N];
int cnt=1, h[N], ver[M<<1], nxt[M<<1];
bool v[N];
void add(int x,int y) { ver[++cnt]=y, nxt[cnt]=h[x], h[x]=cnt; }
void tarjan(int x,int in) {
    R int i, y, z;
    dfn[x]=low[x]=++num;
    for(i=h[x];i;i=nxt[i]) {
        y=ver[i];
        if(!dfn[y]) {
            tarjan(y,i);
            low[x]=min(low[x],low[y]);
            if(dfn[x]<low[y]) v[i]=v[i^1]=1;
        } else if(i!=(in^1)) low[x]=min(low[x],dfn[y]);
    }
}
void dfs(int x) {
    R int i, y;
    c[x]=dcc;
    for(i=h[x];i;i=nxt[i]) {
        y=ver[i];
        if(c[y]!=0||v[i]) continue ;
        dfs(y);
    }
}
int main() {
    R int i, x, y;
    scanf("%d%d",&n,&m);
    for(i=1;i<=m;++i) {
        scanf("%d%d",&x,&y);
        add(x,y), add(y,x);
    }
    tarjan(1,0);
    for(i=1;i<=n;++i) if(!c[i]) ++dcc, dfs(i);
    for(i=2;i<=cnt;++i) {
        x=ver[i^1], y=ver[i];
        if(c[x]!=c[y]) ++deg[c[x]];
    }
    for(i=1;i<=dcc;++i) if(deg[i]==1) ++ans;
    printf("%d\n",(ans+1)>>1);
}
