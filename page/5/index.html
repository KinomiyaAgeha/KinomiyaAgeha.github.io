luogu2480 古代猪文 题解

发表于 2021-08-25 更新于 2021-10-17 分类于 题解 阅读次数：

link

求
$$
\large g^{\sum_{k \mid n}C_n^k} \bmod 9999911659
$$
显然的，那一长串的东西是质数。

设它为 $\large p$ 好了。

则有 $\large \gcd{(g,n)}=1$

显然的，根据欧拉定理的推论可得
$$
\large g^{\sum_{k \mid n}{C_n^k}} \equiv g^{\sum_{k \mid n}{C_n^k , \bmod , p-1}} \quad (\bmod , p)
$$
所以本题就变成了求右边的那个东西。

不难发现但很难算出
$$
\large p-1=2 \times 3 \times 4679 \times 35617
$$
可以用 $\large O(\sqrt{n})$ 的时间枚举 $\large n$ 的约数。

考虑到 $\large n,g \le 10^9$，我们用 $\large lucas$ 定理求出 $\large \sum_{k \mid n}{C_n^k}$，并记录它们对上述四个因子取模的结果，这样就得到了一个线性同余方程组。
$$
\begin{cases}
\large x \bmod 2 = a_1 \
\large x \bmod 3 = a_2 \
\large x \bmod 4679 = a_3 \
\large x \bmod 35617=a_4
\end{cases}
$$
然后用中国剩余定理求出满足上述条件的最小的 $\large x$，用快速幂求 $\large g^x$ 即可。

注意代码细节。

#include<cstdio>
#include<iostream>
using namespace std;
#define R register
#define ll long long
const ll a[4]={2,3,4679,35617};
ll n, g, ans, p=999911658ll, b[4], fac[36000]={1};
ll fp(ll x,ll y) {
    R ll z=1;
    for(;y;x=x*x%p,y>>=1ll) if(y&1ll) z=z*x%p;
    return z;
}
void exgcd(ll a,ll b,ll& x,ll& y) {
    if(!b) { x=1, y=0; return ; }
    exgcd(b,a%b,y,x), y-=a/b*x;
}
ll inv(ll a,ll p) {
    ll x, y;
    exgcd(a,p,x,y);
    return (x%p+p)%p;
}
ll C(ll x,ll p) {
    ll z=1, y=n, a, b;
    for(;x;x/=p,y/=p) {
        a=x%p, b=y%p;
        z=z*fac[b]%p*inv(fac[a],p)%p*inv(b<a?0:fac[b-a],p)%p;
    }
    return z;
}
int main() {
    int i, j;
    ll x, y;
    scanf("%lld%lld",&n,&g), g%=p+1;
    if(!g) { puts("0"); return 0; }
    for(i=1;i<=a[3];++i) fac[i]=fac[i-1]*i%p;
    for(i=1;i*i<=n;++i) if(n%i==0) {
        for(j=0;j<4;++j) b[j]=(b[j]+C(i,a[j]))%a[j];
        if(i*i!=n) for(j=0;j<4;++j) b[j]=(b[j]+C(n/i,a[j]))%a[j];
    }
    for(i=0;i<4;++i) {
        exgcd(p/a[i],a[i],x,y);
        ans=(ans+x*(p/a[i])%p*b[i])%p;
    }
    ans=(ans+p)%p, ++p;
    printf("%lld\n",fp(g,ans));
}

Redundant Paths

初看觉得有些奇怪，仔细思考不难发现就是 Tarjan 算法求出 e-DCC，缩点后统计叶子节点的数量 $\large ans$，$\large (ans+1) \div 2$ 即为答案。

#include<cstdio>
#include<iostream>
using namespace std;
#define R register
const int N=5005, M=10005;
int n, m, num, dcc, ans, dfn[N], low[N], c[N], deg[N];
int cnt=1, h[N], ver[M<<1], nxt[M<<1];
bool v[N];
void add(int x,int y) { ver[++cnt]=y, nxt[cnt]=h[x], h[x]=cnt; }
void tarjan(int x,int in) {
    R int i, y, z;
    dfn[x]=low[x]=++num;
    for(i=h[x];i;i=nxt[i]) {
        y=ver[i];
        if(!dfn[y]) {
            tarjan(y,i);
            low[x]=min(low[x],low[y]);
            if(dfn[x]<low[y]) v[i]=v[i^1]=1;
        } else if(i!=(in^1)) low[x]=min(low[x],dfn[y]);
    }
}
void dfs(int x) {
    R int i, y;
    c[x]=dcc;
    for(i=h[x];i;i=nxt[i]) {
        y=ver[i];
        if(c[y]!=0||v[i]) continue ;
        dfs(y);
    }
}
int main() {
    R int i, x, y;
    scanf("%d%d",&n,&m);
    for(i=1;i<=m;++i) {
        scanf("%d%d",&x,&y);
        add(x,y), add(y,x);
    }
    tarjan(1,0);
    for(i=1;i<=n;++i) if(!c[i]) ++dcc, dfs(i);
    for(i=2;i<=cnt;++i) {
        x=ver[i^1], y=ver[i];
        if(c[x]!=c[y]) ++deg[c[x]];
    }
    for(i=1;i<=dcc;++i) if(deg[i]==1) ++ans;
    printf("%d\n",(ans+1)>>1);
}

这题算是树形 DP 入门题。

树形 DP 的状态设计是统计子树信息，这题我们照葫芦画瓢。

由题意易得每个点有且仅有放与不放两种选择，并且每个点放与不放有后效性，所以加一维专门记录放或不放。

于是就有了设 f(x,0/1)\large f(x,0/1)f(x,0/1) 为以 xxx 为根的子树，xxx 点放与不放所需要的士兵个数。

接着考虑转移。

设 Vx\large V_xV​x​​ 为 x\large xx 点的子节点集合。

当 x\large xx 点不放士兵时，它的子节点必须放。

f(x,0)=∑y∈Vf(y,1)\large f(x,0)=\sum_{y \in V}{f(y,1)} f(x,0)=​y∈V​∑​​f(y,1)

当 xxx 点放士兵时，它的子节点放不放都可以，显然要取较小值。

f(x,1)=∑y∈Vmin{f(y,0),f(y,1)}\large f(x,1)=\sum_{y \in V} \min{\{f(y,0),f(y,1)\}} f(x,1)=​y∈V​∑​​min{f(y,0),f(y,1)}

初始值

\large f(x,0)=0，f(x,1)=1

目标和答案

\large \min{\{f(root，0),f(root,1)\}}

本题读入比较奇怪，用结构体存比较简单。

没有给出根节点，所以读入时把每个子节点打打上标记，最后没有标记的节点即为根。

转移过程由记忆化搜索实现。

 

code

#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
#define R register
const int N=1510;
int n, f[N][2];
bool a[N];
struct pt { int c, s[N]; } g[N];
void dp(int x) {
    R int i, j;
    f[x][0]=0, f[x][1]=1;
    if(!g[x].c) return ;
    for(i=1;i<=g[x].c;++i) {
        dp(g[x].s[i]);
        f[x][0]+=f[g[x].s[i]][1], f[x][1]+=min(f[g[x].s[i]][0],f[g[x].s[i]][1]);
    }
}
void sol() {
    R int i, j, x, y, root=0;
    memset(f,0,sizeof(f)), memset(a,0,sizeof(a));
    for(i=1;i<=n;++i) {
        scanf("%d",&x), scanf(":(%d)",&g[x].c);
        for(j=1;j<=g[x].c;++j) scanf("%d",&y), g[x].s[j]=y, a[y]=1;
    }
    for(;a[root];++root);
    dp(root);
    printf("%d\n",min(f[root][0],f[root][1]));
}
int main() {
    for(;scanf("%d",&n)!=EOF;sol());
}

洛谷2862 Corral the Cows 题解

几个月前写的题解，来水了……

最小化满足条件的正方形边长，正方形边长具有单调性，显然二分答案。

坐标范围为 $[1,10000]$，明显开不下这么大的二维数组。并且 $n$ 很小，不离散化简直是浑身难受啊。

先对读入的数据进行离散化，然后求出二维前缀和。

最后二分一个 $mid$ 表示正方形边长为 $mid$ 是否能满足条件。

判断的时候枚举目标点的横纵坐标，用二维前缀和减去不在正方形内的部分，判断是否大于等于 $c$ 即可。

一定要注意枚举正方形时的边界问题，具体细节看代码。

#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
#define R register
const int N=1010;
int c, n, m, cnt, b[N], s[N][N];
struct pt { int x, y; } a[510];
int f(int x) { return lower_bound(b+1,b+m+1,x)-b; }
bool c_(int w) {
    R int i, j, t;
    for(R int x1=1,x2=1;x1<=m;++x1) {
        for(;b[x2]-b[x1]+1<=w && x2<=m;++x2);
        --x2;
        for(R int y1=1,y2=1;y1<=m;++y1) {
            for(;b[y2]-b[y1]+1<=w && y2<=m;++y2);
            --y2;
            t=s[x2][y2]-s[x1-1][y2]-s[x2][y1-1]+s[x1-1][y1-1];
            if(t>=c) return 1;
        }
    }
    return 0;
}
int main() {
    R int i, j, x, y;
    for(scanf("%d%d",&c,&n),i=1;i<=n;++i) {
        scanf("%d%d",&x,&y);
        a[i]=(pt){x,y}, b[++cnt]=x, b[++cnt]=y;
    }
    sort(b+1,b+cnt+1);
    m=unique(b+1,b+cnt+1)-b-1;
    for(i=1;i<=n;++i) {
        ++s[f(a[i].x)][f(a[i].y)];
    }
    for(i=1;i<=m;++i) for(j=1;j<=m;++j)
        s[i][j]+=s[i-1][j]+s[i][j-1]-s[i-1][j-1];   
    R int l=1, r=1e4, h;
    while(l<r) if(c_(h=l+r>>1)) r=h; else l=h+1;
    printf("%d",l);
}

设 f(x)\large f(x)f(x) 为选择 x\large xx 点的最小代价，令 f(1)=d(1)f(1)=d(1)f(1)=d(1)。

f(y)=f(x)−ct(y)×z+(S−ct(y))×z\large f(y)=f(x)-ct(y) \times z + (S-ct(y)) \times z f(y)=f(x)−ct(y)×z+(S−ct(y))×z

实现上的问题只有子树的点权和无法方便计算，可以用一个变量记录 (x→y)\large (x \rightarrow y)(x→y) 的每个 y\large yy 子树点权和，再用另一个变量将它们累加，最后在遍历子节点后加上自身的点权就能求出完整的 ct\large ctct 数组。

#include<cstdio>
#include<iostream>
using namespace std;
#define R register
#define ll long long
const int N=1e5+10;
ll n, s, c[N];
ll ans=(1ll<<60), ct[N], d[N], f[N];

void add(ll x,ll y,ll z) { ver[++cnt]=y, w[cnt]=z, nxt[cnt]=h[x], h[x]=cnt; }
ll dp(ll x,ll fr) {
    ll res=0;
    R ll i, y, z;
    for(i=h[x];i;i=nxt[i]) if(ver[i]!=fr) {
       y=ver[i], z=w[i];
       ll o=dp(y,x);
       d[x]+=d[y]+z*o, res+=o;
    }
    return ct[x]=res+c[x];
}
void dfs(ll x,ll fr) {
    R ll i, y, z;
    for(i=h[x];i;i=nxt[i]) if(ver[i]!=fr) {
        y=ver[i], z=w[i];
        f[y]=f[x]-((ct[y]*z)<<1ll)+s*z;
        dfs(y,x);
    }
}
int main() {

    for(i=1;i<n;++i) {
        scanf("%lld%lld%lld",&x,&y,&z);
        add(x,y,z), add(y,x,z);
    }

    f[1]=d[1];

给出的是一个内向基环树森林。

很容易想到，如果两个点不在同一颗基环树中，则不存在合法的 $\large (x,y)$。

但是怎么处理在同一颗树中的点呢？

我有一个大胆的想法。

无法直接处理 lca，是因为基环树有一个环。

但是如果 $\large (a,b)$ 两点在同一颗子树中，且该子树没有环上的边，则不影响求出 $\large LCA(a,b)$。
所以，该子树应含不超过1个的环上的点。若含有，则一定是根。

更进一步地说，把每个环上的点看作「根」，它的子树中节点可以直接求 lca，这要我们求出它子树中每个节点的「深度」。

所以，如果 $\large (a,b)$ 两点的「根」是同一个节点，那么只需要求出 $\large c=LCA(a,b)$，$\large dep(a)-dep(c)$ 与 $\large dep(b)-dep(c)$ 即为答案。

若 $\large (a,b)$ 的「根」不同，那么该怎么处理呢？

仔细分析不难发现，在合法的解中，一定有它们与「根」的距离，即 $\large dep(a)-1$ 与 $\large dep(b)-1$ ，记为 $\large (p,q)$

而剩下的部分，就是它们的「根」在环中互相到达的距离。

我们设 $\large ord(x)$ 为环上节点 $x$ 在环上的顺序编号， $\large s(x)$ 为基环树 $\large x$ 环上的节点数量。

设 $\large (x,y)$ 为 $\large (a,b)$ 两点分别的「根」，$\large z$ 为它们所属的基环树编号。

不难发现，有两种情况。

$\large (x \rightarrow y)$ 距离为 $\large ( ord(y)-ord(x)+s(z)) \bmod s(z)$。

$\large (y \rightarrow x)$ 距离为 $\large (ord(x)-ord(y)+s(z)) \bmod s(z)$。

最后检查 $\large (p+(x \rightarrow y),q)$ 与 $(\large p,q+(y \rightarrow x))$，选取合法解。

 

那么如何实现呢？

先找环，每个环对应着每颗基环树。

标记环上的点并将其深度置为1，记录它所在的基环树编号、此树环上节点个数、环上节点的编号、将其根标记为自己。

$\large c(x)$ 用来判断环，$\large d(x)$ 记录深度，$\large bel(x)$ 记录所属基环树编号，$\large root(x)$ 记录环上的「根」。

相信大家都能秒推这道题的方程。

状态：设 $\large f(i,j)$ 为原序列中前 $i$ 个数选出了 $j$ 个的方案数。

转移是显然的。
$$
\large f(i,j)=
\begin{cases}
\large f(i-1,j)+f(i-1,j-1) \quad j \mid a_i\
f(i-1,j) \quad j \nmid a_i
\end{cases}
$$
然而这样的复杂度过高，数组也开不下。

考虑只有 $j \mid a_i$ 时状态才会被更新，否则只会将上一个 $i$ 的状态继承过来。

所以对于每个 $a_i$，能够影响方案数的仅仅只有它的约数。

将每个 $a_i$ 分解，决策集合为 $a_i$ 的约数集合。

然后用滚动数组优化掉第一维。

设 $\large f(i)$ 为长度为 $i$ 的序列的方案数，由于少了一维，所以转移时要累加所有状态，同时也不需要从第 $i-1$ 个数继承状态了。

但是我这样的菜比无法写出优化后的转移方程。

设 $j$ 为 $a_i$ 的约数，将 $\large f(j-1)$ 累加到 $\large f(j)$ 中。

大概就是这样。

 

从「 $a_i$ 的约数 $j$ 对长度为 $j$ 的序列的方案数产生的贡献」考虑的话，我很难明白转移。

所以从 $a_i$ 产生的贡献考虑。

在原来朴素的转移方程中，对于 $a_i$，只会在每个 $j$ 处更新状态，它对总方案数的贡献是更新的这一部分。

这样新状态的转移就很好理解了。

 

这种直接省略一维的滚动数组要注意循环顺序，确保决策从上一轮转移而来。

最后累加答案。

这道题启示我滚动数组优化仅仅是节省了空间，求解问题仍要从原状态、决策、方程考虑。DP 问题的求解要抓住并分析问题的本质。

luogu3288 方伯伯运椰子 题解

发表于 2021-08-19 更新于 2021-10-17 分类于 题解 阅读次数：

link

写一篇不从网络流的角度考虑，单纯推式子的题解。

看到题目给出一张图并要求最大化
$$
\large \frac{X-Y}{k}
$$
就可以确定本题的大致模型：分数规划和 SPFA 判负环。

每条路满负荷，这就意味着我们可以在「理想状态」下进行推导。

设第 $i$ 条边未修改是费用为 $x_i$，修改后费用为 $y_i$。

题目给出 $a_i,b_i,c_i,d_i$，不难想到
$$
\large x_i = c_i \times d_i \implies X = \sum_{i=1}^m {c_i \times d_i}
$$
有两种操作

  • 选择一条道路，将其进行一次压缩，这条道路的容量会下降 1 单位。
  • 选择一条道路，将其进行一次扩容，这条道路的容量会上升 1 单位。

简单推导一下式子，易得当第 $i$ 条边进行压缩时
$$
\large y_i= (c_i-1) \times d_i - a_i \implies Y= \sum_{i=1}^m { (c_i-1) \times d_i + a_i}
$$

进行扩容时
$$
\large y_i = (c_i+1) \times d_i -b_i \implies Y= \sum_{i=1}^m { (c_i+1) \times d_i + b_i }
$$
我们令
$$
\large \max{ { \frac{X-Y}{k} } } = ans
$$
则
$$
\large \frac{X-Y}{k} \large \le ans \
\large X-Y \le \large ans \times k \
\large ans \times k + Y - X \ge \large 0
$$
而
$$
\large Y-X = \sum_{i=1}^m{y_i-x_i} \
\large = \sum_{i=1}^m {\left[(c_i-1) \times d_i + a_i \right]} - \sum_{i=1}^m{c_i \times d_i}\
\large= \sum_{i=1}^m { (c_i \times d_i -d_i +a_i )-c_i \times d_i} \
\large= \sum_{i=1}^m {a_i-d_i} \
\large or \
\large =\sum_{i=1}^m {\left[(c_i+1) \times d_i + b_i \right]} - \sum_{i=1}^m{c_i \times d_i}\
\large =\sum_{i=1}^m{(c_i \times d_i+ d_i +b_i)-c_i \times d_i}\
\large= \sum_{i=1}^m{b_i+d_i}
$$
若将 $y_i-x_i$ 看作图上的边权，修改次数 $k$ 即为某个环上的边数。

所以我们二分一个 $dlt$，判断是否有负环，决定式子是否成立，则分数式又可化为
$$
\large dlt \times k + Y - X \ge \large 0 \
\large \sum_{i=1}^k y_i-x_i+dlt \ge \large 0
$$
由于有两种情况，所以建双向边。

因为原分数式中 $k$ 是分母，所以这个满足上式的 $dlt$ 应尽量小。

然后 SPFA 判负环就行。

若有负环，则不满足上式，令 $l=dlt$，反之则令 $r=dlt$。

#include<bits/stdc++.h>

#define R register
#define ll long long
const int N=5e4+10;
const double eps=1e-4;
int n, m;

bool v[N];
void add(int x,int y,double z) { ver[++c]=y, g[c]=z, nxt[c]=h[x], h[x]=c; }
bool spfa(int x,double dlt) {
    R int i, y, z;
    v[x]=1;
    for(i=h[x];i;i=nxt[i]) {
        y=ver[i], z=g[i];
        if(d[y]>d[x]+z+dlt) {
            d[y]=d[x]+z+dlt;
            if(v[y]||spfa(y,dlt)) return 1;
        }
    }
    return v[x]=0;
}
bool C(double x) {
    memset(v,0,sizeof(v)), memset(d,0,sizeof(d));
    for(R int i=1;i<=n+2;++i) if(spfa(i,x)) return 1;
    return 0;
}

    R double l=0, r=0, mid;
    scanf("%d%d",&n,&m);
    for(i=1;i<=m;++i) {
        scanf("%d%d%d%d%d%d",&x,&y,&A,&B,&P,&Q);
        if(P) add(y,x,(double)(A-Q));
        add(x,y,(double)(B+Q));
        r+=A+B+(double)P*Q;
    }
    while(r-l>eps) {
        mid=(l+r)/2;
        if(C(mid)) l=mid; else r=mid;
    }
    printf("%.2lf",l);
}
