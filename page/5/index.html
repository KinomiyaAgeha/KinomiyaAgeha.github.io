MooFest

两个树状数组维护就行了，细节较多。

#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
#define R register
#define ll long long
const int N=5e4+10;
int n;
ll ans, c[2][N];
struct E { int x, y; } a[N];
bool cmp(E a,E b) { return a.y<b.y; }
void modify(int x,ll y,int z) { for(;x<=5e4;x+=x&-x) c[z][x]+=y; }
ll query(int x,int z) {
    R ll y=0;
    for(;x;x-=x&-x) y+=c[z][x];
    return y;
}
int main() {
    R int i;
    R ll x, y, z=0;
    scanf("%d",&n);
    for(i=1;i<=n;++i) scanf("%d%d",&a[i].y,&a[i].x);
    sort(a+1,a+n+1,cmp);
    for(i=1;i<=n;++i) {
        x=query(a[i].x,0), y=query(a[i].x,1);
        ans+=(x*a[i].x-y)*a[i].y;
        ans+=((z-y)-(i-x-1)*a[i].x)*a[i].y;
        modify(a[i].x,1,0), modify(a[i].x,a[i].x,1);
        z+=a[i].x;
    }
    printf("%lld\n",ans);
}

#include<cstdio>
#include<iostream>
using namespace std;
#define R register
const int N=5e5+10;
int n, m, s, p, num, k, scc, w[N], sw[N], dfn[N], low[N], stk[N];
int ans, c[N], f[N];
int cnt, h[N], ver[N], nxt[N];
int tc, hc[N], vc[N], nc[N];
bool v[N], b[N], B[N];
void add(int x,int y) { ver[++cnt]=y, nxt[cnt]=h[x], h[x]=cnt; }
void cad(int x,int y) { vc[++tc]=y, nc[tc]=hc[x], hc[x]=tc; }
void tarjan(int x) {
    R int i, y;
    dfn[x]=low[x]=++num, stk[++k]=x, v[x]=1;
    for(i=h[x];i;i=nxt[i]) {
        y=ver[i];
        if(!dfn[y]) {
            tarjan(y);
            low[x]=min(low[x],low[y]);
        } else if(v[y]) low[x]=min(low[x],dfn[y]);

    if(dfn[x]==low[x]) {

        do {
            y=stk[k--], v[y]=0, c[y]=scc;
            sw[scc]+=w[y];
            if(b[y]) B[scc]=1;
        } while(x!=y);
    }
}
void X() {
    R int i, x, y;
    for(x=1;x<=n;++x) for(i=h[x];i;i=nxt[i]) {
        y=ver[i];
        if(c[x]!=c[y]) cad(c[x],c[y]);
    }
}
void dp() {
    R int i, x, y, z;

            y=vc[i];
            f[y]=max(f[y],f[x]);
        }
    }

#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
#define R register
#define SET(x,y) memset(x,y,sizeof(x))
#define rep(i,j,k) for(i=(j);i<=(k);++i)
const int N=260;
int n, m, q, d[N][N], dis[N][N], rk[N];
struct E { int z, id; } c[N];
bool cmp(E a,E b) { return a.z!=b.z? a.z<b.z:a.id<b.id; }
int main() {
    R int i, j, k, x, y, z;
    SET(d,0x3f), SET(dis,0x3f);
    scanf("%d%d%d",&n,&m,&q);
    rep(i,1,n) scanf("%d",&c[i].z), c[i].id=i, d[i][i]=0;
    sort(c+1,c+n+1,cmp);
    rep(i,1,n) rk[c[i].id]=i;
    while(m --> 0) {
        scanf("%d%d%d",&x,&y,&z);
        d[rk[x]][rk[y]]=d[rk[y]][rk[x]]=min(d[rk[x]][rk[y]],z);
    }
    rep(k,1,n) rep(i,1,n) rep(j,1,n) if(i!=j) {
        d[i][j]=min(d[i][j],d[i][k]+d[k][j]);
        dis[i][j]=min(dis[i][j],d[i][j]+max(c[k].z,max(c[i].z,c[j].z)));
    }
    while(q --> 0) scanf("%d%d",&x,&y), printf("%d\n",dis[rk[x]][rk[y]]);
}

 

T7

Redundant Paths

初看觉得有些奇怪，仔细思考不难发现就是 Tarjan 算法求出 e-DCC，缩点后统计叶子节点的数量 ans\large ansans，(ans+1)÷2\large (ans+1) \div 2(ans+1)÷2 即为答案。

#include<cstdio>
#include<iostream>
using namespace std;
#define R register
const int N=5005, M=10005;
int n, m, num, dcc, ans, dfn[N], low[N], c[N], deg[N];
int cnt=1, h[N], ver[M<<1], nxt[M<<1];
bool v[N];
void add(int x,int y) { ver[++cnt]=y, nxt[cnt]=h[x], h[x]=cnt; }
void tarjan(int x,int in) {
    R int i, y, z;
    dfn[x]=low[x]=++num;
    for(i=h[x];i;i=nxt[i]) {
        y=ver[i];
        if(!dfn[y]) {
            tarjan(y,i);
            low[x]=min(low[x],low[y]);
            if(dfn[x]<low[y]) v[i]=v[i^1]=1;
        } else if(i!=(in^1)) low[x]=min(low[x],dfn[y]);
    }
}
void dfs(int x) {
    R int i, y;
    c[x]=dcc;
    for(i=h[x];i;i=nxt[i]) {
        y=ver[i];
        if(c[y]!=0||v[i]) continue ;
        dfs(y);
    }
}
int main() {
    R int i, x, y;
    scanf("%d%d",&n,&m);
    for(i=1;i<=m;++i) {
        scanf("%d%d",&x,&y);
        add(x,y), add(y,x);
    }
    tarjan(1,0);
    for(i=1;i<=n;++i) if(!c[i]) ++dcc, dfs(i);
    for(i=2;i<=cnt;++i) {
        x=ver[i^1], y=ver[i];
        if(c[x]!=c[y]) ++deg[c[x]];
    }
    for(i=1;i<=dcc;++i) if(deg[i]==1) ++ans;
    printf("%d\n",(ans+1)>>1);
}

#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
#define R register
const int N=2006;
int n, ans, a[N], b[N];
bool v[N];
bool cmp(int x,int y) { return x>y; }
int main() {
    R int i, j, x, y;
    scanf("%d",&n);
    for(i=1;i<=n;++i) scanf("%d",&a[i]);
    for(i=1;i<=n;++i) scanf("%d",&b[i]);
    sort(a+1,a+n+1,cmp), sort(b+1,b+n+1,cmp);
    for(i=1,j=1,x=n,y=n;i<=x;) {
        if(a[i]>b[j]) ans+=200, ++i, ++j;
        else if(a[i]<b[j]) ans-=200, --x, ++j;
        else {
            if(a[x]>b[y]) ans+=200, --x, --y;
            else {
                if(a[x]<b[j]) ans-=200;
                --x, ++j;
            }
        }
    }
    printf("%d\n",ans);
}

 

T2

爆弹虐场

一开始看错条件了。

「至少 k\large kk 个」看成了「k\large kk 个」。

于是想了一个错误的贪心加边的算法。

看清题目后，发现这题完全可以当最小生成树来作。有一个贪心将边排序的过程。

kruskal 算法结束后判断时候有 k\large kk 个并判断是否联通（是一棵树）即可。

#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
#define R register
const int N=10010;
int n, m, k, tk, ans, l, r=1e6+10, f[N];
struct E { int x, y, z, s; } a[N<<1];
bool cmp(E a,E b) { return a.z!=b.z? a.z<b.z:a.s<b.s; }
bool v[N];
int get(int x) { return x==f[x]? x:f[x]=get(f[x]); }
bool C(int mid) {
    R int i, x, y, z, s, tk=0, e=0;
    for(i=1;i<=n;++i) f[i]=i;
    sort(a+1,a+m+1,cmp);
    for(i=1;i<=m;++i) {
        x=get(a[i].x), y=get(a[i].y), z=a[i].z, s=a[i].s;
        if(min(z,s)>mid) continue ;
        if(x!=y) {
            if(z<=mid) ++tk;
            ++e, f[x]=y;
        }
    }
    return tk>=k&&e==n-1;
}
int main() {
    R int i;
    scanf("%d%d%d",&n,&k,&m);
    for(i=1;i<=m;++i)
        scanf("%d%d%d%d",&a[i].x,&a[i].y,&a[i].z,&a[i].s);
    while(l<=r) {
        int mid=l+r>>1;
        if(C(mid)) r=mid-1, ans=mid; else l=mid+1;
    }
    printf("%d\n",ans);
}
